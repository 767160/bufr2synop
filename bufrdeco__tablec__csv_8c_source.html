26/*!
27 \fn int bufr_read_tablec_csv ( struct bufr_tablec *tc, char *error )
28 \brief Reads a file with table C content (Code table and bit flags) according with csv WMO format
29 \param tc pointer to a target struct \ref bufr_tablec
30 \param error string where to set error if any
31
32 If succeded return 0, otherwise 1
33*/
34int bufr_read_tablec_csv ( struct bufr_tablec *tc, char *error )
35{
36 char *c;
37 FILE *t;
38 int nt;
39 uint32_t ix;
40 size_t i = 0;
41 struct bufr_descriptor desc;
42 char *tk[16];
43 char caux[256], l[CSV_MAXL];
44
45 if ( tc->path[0] == 0 )
46 {
47 return 1;
48 }
49
50 // If we've already readed this table.
51 if ( strcmp ( tc->path, tc->old_path ) == 0 )
52 {
53 return 0; // all done
54 }
55
56 strcpy ( caux, tc->path );
57 memset ( tc, 0, sizeof ( struct bufr_tablec ) );
58 strcpy ( tc->path,caux );
59 if ( ( t = fopen ( tc->path, "r" ) ) == NULL )
60 {
61 sprintf ( error,"Unable to open table C file '%s'\n", tc->path );
62 return 1;
63 }
64
65 // read first line, it is ignored
66 fgets ( l, CSV_MAXL, t );
67
68 while ( fgets ( l, CSV_MAXL, t ) != NULL && i < BUFR_MAXLINES_TABLEC )
69 {
70
71 // Parse line
72 if ( parse_csv_line ( &nt, tk, l ) < 0 || nt != 6 )
73 {
74 sprintf ( error,"Error parsing csv line from table C file '%s'. Found %d fields in line %lu \n", tc->path, nt, i );
75 return 1;
76 }
77
78 // Check if code contains other than non-numeric i.e. 'All' or '-'
79 // In this case, the line is ignored
80 if ( tk[1][0] == 0 || strchr ( tk[1], '-' ) != NULL || strstr ( tk[1], "All" ) != NULL )
81 continue;
82
83 // Key
84 // First we build the descriptor
85 ix = strtoul ( tk[0], &c, 10 );
86 uint32_t_to_descriptor ( &desc, ix );
87 strcpy ( tc->item[i].key, tk[0] );
88 tc->item[i].x = desc.x;
89 tc->item[i].y = desc.y;
90
91 // Integer value
92 tc->item[i].ival = strtoul ( tk[1], &c, 10 );
93
94 // Description
95 c = & tc->item[i].description[0];
96 if ( strlen ( tk[2] ) < BUFR_EXPLAINED_LENGTH )
97 {
98 c += sprintf ( c,"%s", tk[2] );
99 if ( tk[3][0] && ( strlen ( tk[3] ) + strlen ( tc->item[i].description ) ) < BUFR_EXPLAINED_LENGTH )
100 {
101 c += sprintf ( c," %s", tk[3] );
102 if ( tk[4][0] && ( strlen ( tk[4] ) + strlen ( tc->item[i].description ) ) < BUFR_EXPLAINED_LENGTH )
103 {
104 c += sprintf ( c, " %s", tk[4] );
105 }
106 }
107 }
108 else
109 {
110 tk[2][BUFR_EXPLAINED_LENGTH - 1] = '\0';
111 c += sprintf ( c,"%s", tk[2] );
112 }
113
114 if ( tk[5][0] && ( strlen ( tk[5] ) + strlen ( tc->item[i].description ) ) < BUFR_EXPLAINED_LENGTH )
115 {
116 c += sprintf ( c,"%s", tk[5] );
117 }
118
119 if ( tc->num[desc.x] == 0 )
120 {
121 tc->x_start[desc.x] = i; // marc the start
122 }
123 if ( tc->y_ref[desc.x][desc.y] == 0 )
124 {
125 tc->y_ref[desc.x][desc.y] = i - tc->x_start[desc.x]; // marc the position from start of first x
126 }
127 ( tc->num[desc.x] ) ++;
128 //printf("%ld %s %d %d, %ld %ld\n", i, tc->item[i].key, tc->item[i].x, tc->item[i].y, tc->x_start[desc.x], tc->y_ref[desc.x][desc.y]);
129 i++;
130 }
131 fclose ( t );
132 tc->nlines = i;
133 tc->wmo_table = 1;
134 strcpy ( tc->old_path, tc->path ); // store latest path
135 return 0;
136}

139/*!
140 \fn int bufr_find_tablec_csv_index ( size_t *index, struct bufr_tableb *tb, const char *key, uint32_t code )
141 \brief found a descriptor index in a struct \ref bufr_tablec
142 \param index pointer to a size_t where to set the result if success
143 \param tb pointer to struct \ref bufr_tableb where are stored all table B data
144 \param key descriptor string in format FXXYYY
145 \param code value to search in this table/flag
146
147 Return 0 if success, 1 otherwise
148*/
149int bufr_find_tablec_csv_index ( size_t *index, struct bufr_tablec *tc, const char *key, uint32_t code )
150{
151 uint32_t ix;
152 size_t i, i0;
153 char *c;
154 struct bufr_descriptor desc;
155
156 ix = strtoul ( key, &c, 10 );
157 uint32_t_to_descriptor ( &desc, ix );
158 i0 = tc->x_start[desc.x];
159 for ( i = i0 ; i < i0 + tc->num[desc.x] ; i++ )
160 {
161 if ( tc->item[i].x != desc.x )
162 {
163 return 1; // descriptor not found
164 }
165 if ( tc->item[i].y == desc.y && tc->item[i].ival == code )
166 {
167 *index = i;
168 return 0; // found
169 }
170 }
171 return 1; // not found
172}

210 \fn char * bufrdeco_explained_flag_csv_val(char *expl, size_t dim, struct bufr_descriptor *d, unsigned long ival, uint8_t nbits)
211 \brief gets a string with the meaning of a value for a flag table descriptor
212 \param expl string with resulting meaning
213 \param dim max length alowed for \a expl string
214 \param d pointer to the source descriptor
215 \param ival integer value for the descriptos
216 \param nbits uint8_t with the bit extensión of descriptor
217
218 Remember that in FLAG tables for bufr, bit 1 is most significant and N the less one. Bit N only is set to
219 1 when all others are also set to one, i.e. in case of missing value.
220
221 If something went wrong, it returns NULL . Otherwise it returns \a expl
222*/
223char * bufrdeco_explained_flag_csv_val ( char *expl, size_t dim, struct bufr_tablec *tc, struct bufr_descriptor *d,
224 uint64_t ival, uint8_t nbits )
225{
226 char *s;
227 uint64_t test, test0;
228 uint64_t v;
229 size_t i, j;
230
231 if ( tc->num[d->x] == 0 )
232 {
233 //printf("Descriptor %s No encontrado\n", d->c);
234 return NULL;
235 }
236
237 // First line for descriptor
238 i = tc->x_start[d->x] + tc->y_ref[d->x][d->y];
239
240 // init description
241 s = expl;
242 s[0] = '\0';
243 //printf ( "%ld %s %d %d %d %d\n", i, tc->item[i].key, tc->item[i].x, tc->item[i].y, d->x , d->y );
244 for ( j = 0, test0 = 1 ; j < nbits && ( tc->item[i].x == d->x ) && ( tc->item[i].y == d->y ) ; i++ )
245 {
246 v = tc->item[i].ival; // v is the bit number
247 j++;
248
249 // case 0 with meaning
250 if ( v == 0 )
251 {
252 test0 = 1;
253 if ( ival == 0 )
254 {
255 if ( strlen ( expl ) && ( strlen ( expl ) + 1 ) < dim )
256 {
257 s += sprintf ( s, "|" );
258 }
259 if ( strlen ( expl ) + strlen ( tc->item[i].description ) < dim )
260 {
261 s += sprintf ( s,"%s", tc->item[i].description );
262 }
263 return expl;
264 }
265 }
266
267 test = test0 << ( nbits - v );
268
269 if ( v && ( test & ival ) != 0 )
270 {
271 if ( strlen ( expl ) && ( strlen ( expl ) + 1 ) < dim )
272 {
273 s += sprintf ( s, "|" );
274 }
275 if ( strlen ( expl ) + strlen ( tc->item[i].description ) < dim )
276 {
277 s += sprintf ( s,"%s", tc->item[i].description );
278 }
279 }
280 else
281 {
282 continue;
283 }
284 //printf ( "%s\n", expl );
285 }
286 // if match then we have finished the search
287 return expl;
