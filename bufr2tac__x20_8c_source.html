26 /*!

118 /*!
119  \fn char * vism_to_VV ( char *target, double V )
120  \brief Convert horizontal visibilty in meters to a VV string
121  \param V the visibility (m)
122  \param target the resulting VV string
123 */
124 char * vism_to_VV ( char *target, double V )
125 {
126  if ( V < 100.0 )
127  strcpy ( target, "00" );
128  else if ( V <= 5000.0 )
129  sprintf ( target, "%02d", ( int ) ( V + 0.1 ) / 100 );
130  else if ( V < 6000.0 )
131  sprintf ( target, "50" ); // this is to avoid 51-55 range
132  else if ( V <= 30000.0 )
133  sprintf ( target, "%02d", ( int ) ( V + 0.1 ) / 1000 + 50 );
134  else if ( V <= 70000.0 )
135  sprintf ( target, "%02d", ( int ) ( V - 30000.0 ) / 5000 + 80 );
136  else
137  strcpy ( target, "89" );
138  return target;
139 }

141 /*!
142  \fn int syn_parse_x20 ( struct synop_chunks *syn, struct bufr2tac_subset_state *s )
143  \brief Parse a expanded descriptor with X = 20
144  \param syn pointer to a struct \ref synop_chunks where to set the results
145  \param s pointer to a struct \ref bufr2tac_subset_state where is stored needed information in sequential analysis
146 
147  It returns 0 if success, 1 if problems when processing. If a descriptor is not processed returns 0 anyway
148 */
149 int syn_parse_x20 ( struct synop_chunks *syn, struct bufr2tac_subset_state *s )
150 {
151 
152  switch ( s->a->desc.y )
153  {
154  case 1: // 0 20 001 . Horizontal visibility
155  if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
156  return 0;
157  vism_to_VV ( syn->s1.VV, s->val );
158  break;
159  case 3: // 0 20 003 . Present weather
160  if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
161  {
162  s->mask |= ( SUBSET_MASK_HAVE_NO_SIGNIFICANT_WW | SUBSET_MASK_HAVE_NO_SIGNIFICANT_W1 | SUBSET_MASK_HAVE_NO_SIGNIFICANT_W2 );
163  return 0;
164  }
165  if ( s->ival < 100 )
166  {
167  if ( syn->s1.ix[0] == '/' )
168  strcpy ( syn->s1.ix,"1" );
169  sprintf ( syn->s1.ww, "%02d", s->ival );
170  syn->mask |= SYNOP_SEC1;
171  }
172  else if ( s->ival == 100 )
173  {
174  s->mask |= SUBSET_MASK_HAVE_NO_SIGNIFICANT_WW;
175  }
176  else if ( s->ival < 200 )
177  {
178  if ( syn->s1.ix[0] == '/' )
179  strcpy ( syn->s1.ix,"7" );
180  sprintf ( syn->s1.ww, "%02d", s->ival % 100 );
181  syn->mask |= SYNOP_SEC1;
182  }
183  else if ( s->ival == 508 )
184  {
185  s->mask |= ( SUBSET_MASK_HAVE_NO_SIGNIFICANT_WW | SUBSET_MASK_HAVE_NO_SIGNIFICANT_W1 | SUBSET_MASK_HAVE_NO_SIGNIFICANT_W2 );
186  }
187  break;
188  case 4: // 0 20 004 . Past weather (1)
189  if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
190  {
191  s->mask |= SUBSET_MASK_HAVE_NO_SIGNIFICANT_W1;
192  return 0;
193  }
194  if ( s->ival < 10 )
195  {
196  if ( syn->s1.ix[0] == '/' )
197  strcpy ( syn->s1.ix,"1" );
198  sprintf ( syn->s1.W1, "%d", s->ival );
199  syn->mask |= SYNOP_SEC1;
200  }
201  else if ( s->ival == 10 )
202  {
203  s->mask |= SUBSET_MASK_HAVE_NO_SIGNIFICANT_W1;
204  }
205  else
206  {
207  if ( syn->s1.ix[0] == '/' )
208  strcpy ( syn->s1.ix,"7" );
209  sprintf ( syn->s1.W1, "%d", s->ival % 10 );
210  syn->mask |= SYNOP_SEC1;
211  }
212  break;
213  case 5: // 0 20 005 . Past weather (2)
214  if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
215  {
216  s->mask |= SUBSET_MASK_HAVE_NO_SIGNIFICANT_W2;
217  return 0;
218  }
219  if ( s->ival < 10 )
220  {
221  if ( syn->s1.ix[0] == '/' )
222  strcpy ( syn->s1.ix,"1" );
223  sprintf ( syn->s1.W2, "%d", s->ival );
224  syn->mask |= SYNOP_SEC1;
225  }
226  else if ( s->ival == 10 )
227  {
228  s->mask |= SUBSET_MASK_HAVE_NO_SIGNIFICANT_W2;
229  }
230  else
231  {
232  if ( syn->s1.ix[0] == '/' )
233  strcpy ( syn->s1.ix,"7" );
234  sprintf ( syn->s1.W2, "%d", s->ival % 10 );
235  syn->mask |= SYNOP_SEC1;
236  }
237  break;
238  case 10: // 0 20 010 . Cloud cover (total)
239  if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
240  return 0;
241  percent_to_okta ( syn->s1.N, s->val );
242  break;
243  case 11: // 0 20 011 . Cloud amount
244  if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
245  return 0;
246  if ( s->clayer == 0 )
247  {
248  if ( s->ival <= 8 )
249  sprintf ( syn->s1.Nh, "%1d", s->ival );
250  else if ( s->ival <= 10 )
251  sprintf ( syn->s1.Nh, "9" );
252  else if ( s->ival == 15 )
253  sprintf ( syn->s1.Nh, "/" );
254  syn->mask |= SYNOP_SEC1;
255  }
256  else
257  {
258  if ( s->ival <= 8 )
259  sprintf ( syn->s3.nub[s->clayer - 1].Ns, "%1d", s->ival );
260  else if ( s->ival <= 10 )
261  sprintf ( syn->s3.nub[s->clayer - 1].Ns, "9" );
262  else if ( s->ival == 15 )
263  sprintf ( syn->s3.nub[s->clayer - 1].Ns, "/" );
264  syn->mask |= SYNOP_SEC3;
265  }
266  break;
267  case 12: // 0 20 012 . Cloud type
268  if ( s->clayer == 0 )
269  {
270  if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
271  return 0;
272  if ( s->ival >= 10 && s->ival < 20 )
273  sprintf ( syn->s1.Ch, "%1d", s->ival % 10 );
274  else if ( s->ival >= 20 && s->ival < 30 )
275  sprintf ( syn->s1.Cm, "%1d", s->ival % 10 );
276  else if ( s->ival >= 30 && s->ival < 40 )
277  sprintf ( syn->s1.Cl, "%1d", s->ival % 10 );
278  else if ( s->ival == 59 )
279  sprintf ( syn->s1.Nh, "/" );
280  else if ( s->ival == 60 )
281  sprintf ( syn->s1.Ch, "/" );
282  else if ( s->ival == 61 )
283  sprintf ( syn->s1.Cm, "/" );
284  else if ( s->ival == 62 )
285  sprintf ( syn->s1.Cl, "/" );
286  syn->mask |= SYNOP_SEC1;
287  }
288  else
289  {
290  if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
291  {
292  sprintf ( syn->s3.nub[s->clayer - 1].C, "/" );
293  return 0;
294  }
295  if ( s->ival == 59 )
296  sprintf ( syn->s3.nub[s->clayer - 1].C, "/" );
297  else
298  sprintf ( syn->s3.nub[s->clayer - 1].C, "%1d", s->ival % 10 );
299  syn->mask |= SYNOP_SEC3;
300  }
301 
302  break;
303  case 13: // 0 20 013 . Height of base of cloud
304  if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
305  return 0;
306  if ( s->clayer == 0 ) // fisrt layer is for sec1
307  m_to_h ( syn->s1.h, s->val );
308  else
309  m_to_hh ( syn->s3.nub[s->clayer - 1].hshs, s->val );
310  break;
311  case 62: // 0 20 062 . State of the ground (with or without snow)
312  if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
313  return 0;
314  if ( s->ival < 10 )
315  sprintf ( syn->s3.E,"%d", s->ival );
316  else if ( s->ival < 20 )
317  sprintf ( syn->s3.E1,"%d", s->ival % 10 );
318  syn->mask |= SYNOP_SEC3;
319  break;
320  default:
321  break;
322  }
323 
324  // check about h
325  if ( syn->s1.h[0] == '/' && syn->s1.N[0] == '0' )
326  syn->s1.h[0] = '9';
327 
328  // Supress 7wwW1W2 if no significant weather
329  if ( ( syn->s1.ww[0] == 0 || ( syn->s1.ww[0] && ( strcmp ( syn->s1.ww,"04" ) < 0 ) ) ) &&
330  ( syn->s1.W1[0] == 0 || ( syn->s1.W1[0] && ( strcmp ( syn->s1.W1,"3" ) < 0 ) ) ) &&
331  ( syn->s1.W2[0] == 0 || ( syn->s1.W2[0] && ( strcmp ( syn->s1.W2,"3" ) < 0 ) ) ) )
332  {
333  syn->s1.ww[0] = 0;
334  syn->s1.W1[0] = 0;
335  syn->s1.W2[0] = 0;
336  if ( syn->s1.ix[0] == '1' )
337  syn->s1.ix[0] = '2';
338  }
339 
340  return 0;
341 }
