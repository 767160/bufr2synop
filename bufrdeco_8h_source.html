242 /*!
243  \struct bufr_sequence
244  \brief Stores an unexpanded sequence of descriptors
245 
246  A sequence layer is needed when parsing expanded descriptor sec3 and sec4
247 
248  First bufr_sequence is the sequence of descriptors in sec3 after
249  byte 8. This is a bufr_sequence in level 0.
250 
251  When a sequence descriptor is found in a layer, the sequence entries found in table D
252  form this descriptor is a son bufr_sequence. This son has then a father
253  and also can have one or more sons. The index level is incremented by one every step it
254  go into decendents.
255 
256 */
257 struct bufr_sequence
258 {
259  char key[8]; /*!< the key */
260  size_t level; /*!< The recursion level. descriptors in sec3 are level 0 */
261  struct bufr_sequence *father; /*!< Pointer to the father struct. It should be NULL if level = 0 */
262  size_t ndesc; /*!< Number of unexpanded descriptor of a layer */
263  struct bufr_descriptor lseq[NMAXSEQ_DESCRIPTORS]; /*!< Array of unexpanded descriptors */
264  struct bufr_sequence *sons[NMAXSEQ_DESCRIPTORS]; /*!< Array of pointers to sons. It must be NULL
265  except for sequence descriptors. */
266 };

344 /*!
345  \struct bufr_sec1
346  \brief Store de parsed section 1 of a bufr file
347 
348  <pre>
349  This is for version 4 BUFR
350  Section 1 – Identification section
351  Octet No. Contents
352  1–3 Length of section
353  4 BUFR master table (zero if standard WMO FM 94 BUFR tables are used – see Note 2)
354  5–6 Identification of originating/generating centre (see Common Code table C-11)
355  7–8 Identification of originating/generating sub-centre (allocated by originating/generating centre
356  – see Common Code table C-12)
357  9 Update sequence number (zero for original messages and for messages containing only
358  delayed reports; incremented for the other updates)
359  10 Bit 1 = 0 No optional section
360  = 1 Optional section follows
361  Bits 2-8 Set to zero (reserved)
362  11 Data category (Table A)
363  12 International data sub-category (see Common Code table C-13 and Note 3)
364  13 Local data sub-category (defined locally by automatic data-processing (ADP) centres
365  14 Version number of master table
366  15 Version number of local tables used to augment master table in use
367  16–17 Year (4 digits)
368  18 Month
369  19 Day Most typical time for the BUFR message contents – see Note 4
370  20 Hour
371  21 Minute
372  22 Second
373  23–Optional – for local use by ADP centres
374 
375  And this is for version 3
376  1–3 Length of section
377  4 BUFR master table (zero if standard WMO FM 94 BUFR tables are used – see Note 2)
378  5 Originating/generating sub-centre: Code table 0 01 034 (defined, if necessary, by associated
379  originating/generating centre – see Note 3 of Common Code table C–1 in Part C/c.)
380  6 Originating/generating centre: Code table 0 01 033 (Common Code table C–1 in Part C/c.)
381  7 Update sequence number (zero for original BUFR messages; incremented for updates)
382  8 Bit 1 = 0 No optional section
383  = 1 Optional section follows
384  Bits 2–8 Set to zero (reserved)
385  9 Data category (Table A)
386  10 Data sub-category (defined by local automatic data processing (ADP) centres)
387  11 Version number of master table used – see Notes 2 and 4
388  12 Version number of local tables used to augment the master table in use – see Note 2
389  13 Year of century
390  14 Month
391  15 Day Most typical for the BUFR message contents
392  16 Hour
393  17 Minute
394  18- If not required by ADP centres for local use, octet 18 only shall be included and set to zero with
395  reference to Regulation 94.1.3
396  </pre>
397 */

400  uint32_t length; /*!< Length of section 1 */
401  uint8_t master; /*!< Bufr master table */
402  uint32_t centre; /*!< Centre */
403  uint32_t subcentre; /*!< sub-Centre */
404  uint8_t update; /*!< update sequence number */
405  uint8_t options; /*!< options */
406  uint8_t category; /*!< Data category (table A) */
407  uint8_t subcategory; /*!< subcategory */
408  uint8_t subcategory_local;/*!< local data subcategory */
409  uint8_t master_version; /*!< Version of master table */
410  uint8_t master_local; /*!< Version number of local tables used to augment master table in use */
411  uint32_t year; /*!< Year */
412  uint8_t month; /*!< Month */
413  uint8_t day; /*!< Day */
414  uint8_t hour; /*!< hour */
415  uint8_t minute; /*!< minute */
416  uint8_t second; /*!< second */
