307 /*!
308  \struct bufr_sequence
309  \brief Stores an unexpanded sequence of descriptors
310 
311  A sequence layer is needed when parsing expanded descriptor sec3 and sec4
312 
313  First bufr_sequence is the sequence of descriptors in sec3 after
314  byte 8. This is a bufr_sequence in level 0.
315 
316  When a sequence descriptor is found in a layer, the sequence entries found in table D
317  form this descriptor is a son bufr_sequence. This son has then a father
318  and also can have one or more sons. The index level is incremented by one every step it
319  go into decendents.
320 
321 */
322 struct bufr_sequence
323 {
324  char key[8]; /*!< the key */
325  size_t level; /*!< The recursion level. descriptors in sec3 are level 0 */
326  struct bufr_sequence *father; /*!< Pointer to the father struct. It should be NULL if level = 0 */
327  size_t ndesc; /*!< Number of unexpanded descriptor of a layer */
328  struct bufr_descriptor lseq[NMAXSEQ_DESCRIPTORS]; /*!< Array of unexpanded descriptors */
329  struct bufr_sequence *sons[NMAXSEQ_DESCRIPTORS]; /*!< Array of pointers to sons. It must be NULL
330  except for sequence descriptors. */
331 };

397 /*!
398  \struct bufr_sec1
399  \brief Store de parsed section 1 of a bufr file
400 
401  <pre>
402  This is for version 4 BUFR
403  Section 1 – Identification section
404  Octet No. Contents
405  1–3 Length of section
406  4 BUFR master table (zero if standard WMO FM 94 BUFR tables are used – see Note 2)
407  5–6 Identification of originating/generating centre (see Common Code table C-11)
408  7–8 Identification of originating/generating sub-centre (allocated by originating/generating centre
409  – see Common Code table C-12)
410  9 Update sequence number (zero for original messages and for messages containing only
411  delayed reports; incremented for the other updates)
412  10 Bit 1 = 0 No optional section
413  = 1 Optional section follows
414  Bits 2-8 Set to zero (reserved)
415  11 Data category (Table A)
416  12 International data sub-category (see Common Code table C-13 and Note 3)
417  13 Local data sub-category (defined locally by automatic data-processing (ADP) centres
418  14 Version number of master table
419  15 Version number of local tables used to augment master table in use
420  16–17 Year (4 digits)
421  18 Month
422  19 Day Most typical time for the BUFR message contents – see Note 4
423  20 Hour
424  21 Minute
425  22 Second
426  23–Optional – for local use by ADP centres
427 
428  And this is for version 3
429  1–3 Length of section
430  4 BUFR master table (zero if standard WMO FM 94 BUFR tables are used – see Note 2)
431  5 Originating/generating sub-centre: Code table 0 01 034 (defined, if necessary, by associated
432  originating/generating centre – see Note 3 of Common Code table C–1 in Part C/c.)
433  6 Originating/generating centre: Code table 0 01 033 (Common Code table C–1 in Part C/c.)
434  7 Update sequence number (zero for original BUFR messages; incremented for updates)
435  8 Bit 1 = 0 No optional section
436  = 1 Optional section follows
437  Bits 2–8 Set to zero (reserved)
438  9 Data category (Table A)
439  10 Data sub-category (defined by local automatic data processing (ADP) centres)
440  11 Version number of master table used – see Notes 2 and 4
441  12 Version number of local tables used to augment the master table in use – see Note 2
442  13 Year of century
443  14 Month
444  15 Day Most typical for the BUFR message contents
445  16 Hour
446  17 Minute
447  18- If not required by ADP centres for local use, octet 18 only shall be included and set to zero with
448  reference to Regulation 94.1.3
449  </pre>
450 */
