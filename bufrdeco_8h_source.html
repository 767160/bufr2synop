411/*!
412 \struct bufr_sequence
413 \brief Stores an unexpanded sequence of descriptors
414
415 A sequence layer is needed when parsing expanded descriptor sec3 and sec4
416
417 First bufr_sequence is the sequence of descriptors in sec3 after
418 byte 8. This is a bufr_sequence in level 0.
419
420 When a sequence descriptor is found in a layer, the sequence entries found in table D
421 form this descriptor is a son bufr_sequence. This son has then a father
422 and also can have one or more sons. The index level is incremented by one every step it
423 go into decendents.
424
425*/
426struct bufr_sequence
427{
428 char key[8]; /*!< the key */
429 size_t level; /*!< The recursion level. descriptors in sec3 are level 0 */
430 struct bufr_sequence *father; /*!< Pointer to the father struct. It should be NULL if level = 0 */
431 size_t ndesc; /*!< Number of unexpanded descriptor of a layer */
432 struct bufr_descriptor lseq[NMAXSEQ_DESCRIPTORS]; /*!< Array of unexpanded descriptors */
433 struct bufr_sequence *sons[NMAXSEQ_DESCRIPTORS]; /*!< Array of pointers to sons. It must be NULL
434 except for sequence descriptors. */
435 int iseq; /*!< number of sequence when parsing a tree. for recursion level 0 in sec3 is asigned 0.
436 Is the index in member seq in struct bufereco_expanded_tree */
437 char name[BUFR_EXPLAINED_LENGTH]; /*!< Name of sequence if any */
438};
