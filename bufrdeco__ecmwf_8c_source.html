30const char DEFAULT_BUFRTABLES_ECMWF_DIR2[] = "/usr/lib/bufrtables/";

32/*!
33 \fn int get_ecmwf_tablenames ( struct bufrdeco *b, const char *bufrtables_dir)
34 \brief Get the complete pathnames for EMCWF table files needed by a bufr message
35 \param bufrtables_dir string with path to bufr file tables dir
36 \param b pointer for a struct \ref bufrdeco
37
38 In ECMWF library the name of a table file is Kssswwwwwxxxxxyyyzzz.TXT , where
39 - K - type of table, i.e, 'B', 'C', or 'D'
40 - sss - Master table number (zero for WMO meteorological tables)
41 - wwwww - Originating sub-centre
42 - xxxxx - Originating centre
43 - yyy - Version number of master table used
44 - zzz - Version number of local table used
45*/
46int get_ecmwf_tablenames ( struct bufrdeco *b )
47{
48 struct stat st;
49 char aux[128];
50
51 if ( b->bufrtables_dir[0] == '\0' )
52 {
53 // try to guess directory
54 if ( stat ( DEFAULT_BUFRTABLES_ECMWF_DIR1, &st ) )
55 {
56 if ( stat ( DEFAULT_BUFRTABLES_ECMWF_DIR2, &st ) )
57 {
58 return 1;
59 }
60 else
61 {
62 if ( S_ISDIR ( st.st_mode ) )
63 {
64 strcpy ( aux,DEFAULT_BUFRTABLES_ECMWF_DIR2 );
65 }
66 }
67 }
68 else
69 {
70 if ( S_ISDIR ( st.st_mode ) )
71 {
72 strcpy ( aux,DEFAULT_BUFRTABLES_ECMWF_DIR1 );
73 }
74 }
75 }
76 else
77 {
78 strcpy ( aux, b->bufrtables_dir );
79 }
80
81 sprintf ( b->tables->b.path,"%sB%03d%05d%05d%03d%03d.TXT", aux, b->sec1.master,
82 b->sec1.subcentre, b->sec1.centre, b->sec1.master_version, b->sec1.master_local );
83 sprintf ( b->tables->c.path,"%sC%03d%05d%05d%03d%03d.TXT", aux, b->sec1.master,
84 b->sec1.subcentre, b->sec1.centre, b->sec1.master_version, b->sec1.master_local );
85 sprintf ( b->tables->d.path,"%sD%03d%05d%05d%03d%03d.TXT", aux, b->sec1.master,
86 b->sec1.subcentre, b->sec1.centre, b->sec1.master_version, b->sec1.master_local );
87
88 /* check for table b, if problems then we try alternative */
89 if ( stat ( b->tables->b.path, &st ) )
90 {
91 // here we set originating centre xxxxx to 00000 for WMO tables
92 if ( b->sec1.master != 0 ) // case of not WMO tables
93 {
94 sprintf ( b->tables->b.path,"%sB%03d%05d%05d%03d%03d.TXT", aux, b->sec1.master,
95 b->sec1.subcentre, b->sec1.centre, b->sec1.master_version, b->sec1.master_local );
96 sprintf ( b->tables->c.path,"%sC%03d%05d%05d%03d%03d.TXT", aux, b->sec1.master,
97 b->sec1.subcentre, b->sec1.centre, b->sec1.master_version, b->sec1.master_local );
98 sprintf ( b->tables->d.path,"%sD%03d%05d%05d%03d%03d.TXT", aux, b->sec1.master,
99 b->sec1.subcentre, b->sec1.centre, b->sec1.master_version, b->sec1.master_local );
100 }
101 else
102 {
103 sprintf ( b->tables->b.path,"%sB000%05d00000%03d%03d.TXT", aux, b->sec1.subcentre,
104 b->sec1.master_version, b->sec1.master_local );
105 sprintf ( b->tables->c.path,"%sC000%05d00000%03d%03d.TXT", aux, b->sec1.subcentre,
106 b->sec1.master_version, b->sec1.master_local );
107 sprintf ( b->tables->d.path,"%sD000%05d00000%03d%03d.TXT", aux, b->sec1.subcentre,
108 b->sec1.master_version, b->sec1.master_local );
109 }
110
111 if ( stat ( b->tables->b.path, &st ) )
112 {
113 // Another chance. Set local zzz to 000
114 if ( b->sec1.master != 0 ) // case of not WMO tables
115 {
116 sprintf ( b->tables->b.path,"%sB%03d%05d%05d%03d000.TXT", aux, b->sec1.master,
117 b->sec1.subcentre, b->sec1.centre, b->sec1.master_version );
118 sprintf ( b->tables->c.path,"%sC%03d%05d%05d%03d000.TXT", aux, b->sec1.master,
119 b->sec1.subcentre, b->sec1.centre, b->sec1.master_version );
120 sprintf ( b->tables->d.path,"%sD%03d%05d%05d%03d000.TXT", aux, b->sec1.master,
121 b->sec1.subcentre, b->sec1.centre, b->sec1.master_version );
122 }
123 else
124 {
125 sprintf ( b->tables->b.path,"%sB000%05d00000%03d000.TXT", aux, b->sec1.subcentre,
126 b->sec1.master_version );
127 sprintf ( b->tables->c.path,"%sC000%05d00000%03d000.TXT", aux, b->sec1.subcentre,
128 b->sec1.master_version );
129 sprintf ( b->tables->d.path,"%sD000%05d00000%03d000.TXT", aux, b->sec1.subcentre,
130 b->sec1.master_version );
131 }
132
133 if ( stat ( b->tables->b.path, &st ) )
134 {
135 // Another chance. Set subcentre wwwww to 00000
136 if ( b->sec1.master != 0 ) // case of not WMO tables
137 {
138 sprintf ( b->tables->b.path,"%sB%03d%05d%05d%03d000.TXT", aux, b->sec1.master,
139 b->sec1.subcentre, b->sec1.centre, b->sec1.master_version );
140 sprintf ( b->tables->c.path,"%sC%03d%05d%05d%03d000.TXT", aux, b->sec1.master,
141 b->sec1.subcentre, b->sec1.centre, b->sec1.master_version );
142 sprintf ( b->tables->d.path,"%sD%03d%05d%05d%03d000.TXT", aux, b->sec1.master,
143 b->sec1.subcentre, b->sec1.centre, b->sec1.master_version );
144 }
145 else
146 {
147 sprintf ( b->tables->b.path,"%sB0000000000000%03d000.TXT", aux, b->sec1.master_version );
148 sprintf ( b->tables->c.path,"%sC0000000000000%03d000.TXT", aux, b->sec1.master_version );
149 sprintf ( b->tables->d.path,"%sD0000000000000%03d000.TXT", aux, b->sec1.master_version );
150 }
151 }
152 }
153 }
154 return 0;
155}

157/*!
158 \fn int bufr_read_tables_ecmwf (struct bufrdeco *b, char *tables_dir)
159 \brief Read the tables according with bufr file data from a bufr table directory
160 \param b basic struct with needed data
161
162 The default directories where to search bufr tables are stored in \ref DEFAULT_BUFRTABLES_ECMWF_DIR1 and \ref DEFAULT_BUFRTABLES_ECMWF_DIR2
163*/
164int bufr_read_tables_ecmwf ( struct bufrdeco *b )
165{
166
167 // get tablenames
168 if ( get_ecmwf_tablenames ( b ) )
169 {
170 sprintf ( b->error, "bufrdeco_read_tables_ecmwf(): Cannot find bufr tables\n" );
171 return 1;
172 }
173
174 // If tables still not initialized then do it
175 if ( b->tables == NULL && bufrdeco_init_tables ( & ( b->tables ) ) )
176 {
177 sprintf ( b->error, "bufrdeco_read_tables_ecmwf(): Cannot allocate memory for tables\n" );
178 return 1;
179 }
180
181 // And now read tables
182 if ( bufr_read_tableb ( & ( b->tables->b ), b->error ) )
183 {
184 return 1;
185 }
186 if ( bufr_read_tablec ( & ( b->tables->c ), b->error ) )
187 {
188 return 1;
189 }
190 if ( bufr_read_tabled ( & ( b->tables->d ), b->error ) )
191 {
192 return 1;
193 }
194 return 0;
195}
