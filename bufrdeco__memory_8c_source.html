95 /*!
96  \fn int bufrdeco_substitute_tables ( struct bufr_tables **replaced, struct bufr_tables *source, struct bufrdeco *b )
97  \brief substitute an struct \ref bufr_tables into a struct \ref bufrdeco
98  \param replaced Pointer where to set the replaced pointer
99  \param source Pointer to a struct \ref bufr_tables
100  \param b pointer to the container basic struct \ref bufrdeco
101 
102  Remember that the struct \ref bufr_tables used in bufrdeco library is the one which pointer is in struct
103  \ref bufrdeco . To avoid problems the struct must be initialized before substituted in this fucntion.
104  Both source and replaced structs are not modified.
105 
106  This is useful if we do not want to read and parse tables again if the caller has a pool of
107  already readed tables.
108 */
109 int bufrdeco_substitute_tables ( struct bufr_tables **replaced, struct bufr_tables *source, struct bufrdeco *b )
110 {
111  *replaced = b->tables;
112  if ( source == NULL )
113  {
114  // allocate memory for table
115  return bufrdeco_init_tables ( & ( b->tables ) );
116  }
117  else
118  b->tables = source;
119  return 0;
120 }

240 /*!
241  \fn int bufrdeco_init(struct bufrdeco *b)
242  \brief Inits and allocate memory for a struct \ref bufrdeco
243  \param b pointer to the target struct
244 
245  This function only must be called once. When finished the function \ref bufrdeco_close must be called to
246  free all needed memory
247 */
248 int bufrdeco_init ( struct bufrdeco *b )
249 {
250  if ( b == NULL )
251  return 1;
252 
253  // First clear all
254  memset ( b, 0, sizeof ( struct bufrdeco ) );
255 
256  // Then allocate all needed memory when starting. Further we will need to allocate some more depending on
257  // data
258 
259  // allocate memory for Tables
260  if ( bufrdeco_init_tables ( &b->tables ) )
261  {
262  sprintf ( b->error,"init_bufrdeco(): Cannot allocate space for tables\n" );
263  return 1;
264  }
265 
266  // allocate memory for expanded tree of descriptors
267  if ( bufrdeco_init_expanded_tree ( &b->tree ) )
268  {
269  sprintf ( b->error,"init_bufrdeco(): Cannot allocate space for expanded tree of descriptors\n" );
270  return 1;
271  }
272 
273 
274  return 0;
275 }

291  memset ( & ( b->sec3 ), 0, sizeof ( struct bufr_sec3 ) );
292  memset ( & ( b->sec4 ), 0, sizeof ( struct bufr_sec4 ) );

301  \fn int bufrdeco_close ( struct bufrdeco *b )
302  \brief Free all allocated memory

304

307 int bufrdeco_close ( struct bufrdeco *b )
