31/*!
32 \fn char *bufrdeco_get_version(char *version, char *build, char *builder, int *version_major, int *version_minor, int *version_patch)
33 \brief Get strings with version information and build date and time
34 \param version pointer to string with version as result if not NULL
35 \param build pointer to string with compiler and compilation date and time if not NULL
36 \param builder pointer to string with builder utility. 'cmake' or 'autotools' if not NULL
37 \param version_major pointer to string with version_major component if not NULL
38 \param version_minor pointer to string with version_minor component if not NULL
39 \param version_patch pointer to string with version_patch component if not NULL
40
41 Retuns string pointer \ref version.
42 */
43char *bufrdeco_get_version(char *version, char *build, char *builder, int *version_major, int *version_minor, int *version_patch)
44{
45 int major = 0, minor = 0, patch = 0;
46 char *c;
47 if (version == NULL)
48 return NULL;
49 sprintf(version, "%s", VERSION);
50 // default values
51 sscanf(version, "%d.%d.%d", &major, &minor, &patch);
52
53 if (build)
54 {
55 c = build;
56#if defined(__INTEL_COMPILER)
57 c += sprintf(build, "using INTEL C compiler icc %d.%d ", __INTEL_COMPILER, __INTEL_COMPILER_UPDATE);
58#elif defined(__GNUC__)
59 c += sprintf(build, "using GNU C compiler gcc %d.%d.%d ", __GNUC__ , __GNUC_MINOR__ , __GNUC_PATCHLEVEL__);
60#endif
61 sprintf(c,"at %s %s",__DATE__,__TIME__);
62 }
63
64 if (builder)
65#ifdef BUILD_USING_CMAKE
66 strcpy(builder, "cmake");
67#else
68 strcpy(builder, "autotools");
69#endif
70 if (version_major)
71 *version_major = major;
72 if (version_minor)
73 *version_minor = minor;
74 if (version_patch)
75 *version_patch = patch;
76 return version;
77}
