26/*!
27 \fn int bufr_read_tablec ( struct bufr_tablec *tc, char *error )
28 \brief Reads a file with table C content according with ECMWF format
29 \param tc pointer to a target struct \ref bufr_tablec
30 \param error string where to set error if any
31
32 If succeded return 0, otherwise 1
33*/
34int bufr_read_tablec ( struct bufr_tablec *tc, char *error )
35{
36 char aux[8], *c;
37 size_t startx = 0;
38 FILE *t;
39 size_t i = 0;
40 char caux[256];
41
42 if ( tc->path[0] == '\0' )
43 {
44 return 1;
45 }
46
47 // If we've already readed this table.
48 if ( strcmp ( tc->path, tc->old_path ) == 0 )
49 {
50 return 0; // all done
51 }
52
53 strcpy ( caux, tc->path );
54 memset ( tc, 0, sizeof ( struct bufr_tablec ) );
55 strcpy ( tc->path,caux );
56 if ( ( t = fopen ( tc->path, "r" ) ) == NULL )
57 {
58 sprintf ( error,"Unable to open table C file '%s'\n", tc->path );
59 return 1;
60 }
61
62 while ( fgets ( tc->l[i], 96, t ) != NULL && i < BUFR_MAXLINES_TABLEC )
63 {
64 // supress the newline
65 if ( ( c = strrchr ( tc->l[i],'\n' ) ) != NULL )
66 {
67 *c = '\0';
68 }
69 if ( tc->l[i][1] != ' ' && tc->l[i][2] != ' ' )
70 {
71 aux[0] = tc->l[i][1];
72 aux[1] = tc->l[i][2];
73 aux[2] = '\0';
74 startx = strtoul ( aux, &c, 10 );
75 if ( tc->x_start[startx] == 0 )
76 {
77 tc->x_start[startx] = i; // marc the start
78 }
79 }
80 ( tc->num[startx] ) ++;
81 i++;
82 }
83 fclose ( t );
84 tc->nlines = i;
85 tc->wmo_table = 0;
86 strcpy ( tc->old_path, tc->path ); // store latest path
87 return 0;
88}

90/*!
91 \fn int bufr_find_tablec_index ( size_t *index, struct bufr_tablec *tc, const char *key )
92 \brief Find the index of a line in table C for a given key of a descriptor
93 \param index pointer where to set the result
94 \param tc pointer to a struct \ref bufr_tablec where all table data is stored
95 \param key string in the form FXXYYY which is the key of descriptor we want to find out
96
97 If the descriptor has been found with success then returns 0, othewise returns 1
98*/
99int bufr_find_tablec_index ( size_t *index, struct bufr_tablec *tc, const char *key )
100{
101 size_t i, i0;
102 size_t x = 0;
103 char *c, aux[8];
104
105 aux[0] = key[1];
106 aux[1] = key[2];
107 aux[2] = '\0';
108 x = strtoul ( aux, &c, 10 );
109 i0 = tc->x_start[x];
110 for ( i = i0 ; i < i0 + tc->num[x] ; i++ )
111 {
112 if ( tc->l[i][0] != key[0] ||
113 tc->l[i][1] != key[1] ||
114 tc->l[i][2] != key[2] ||
115 tc->l[i][3] != key[3] ||
116 tc->l[i][4] != key[4] ||
117 tc->l[i][5] != key[5] )
118 {
119 continue;
120 }
121 else
122 {
123 *index = i;
124 return 0;
125 }
126 }
127 return 1; // not found
128}

131/*!
132 \fn char * bufrdeco_explained_table_val (char *expl, size_t dim, struct bufr_tablec *tc, struct bufr_descriptor *d, int ival)
133 \brief gets a string with the meaning of a value for a code table descriptor
134 \param expl string with resulting meaning
135 \param dim numero máximo de caracteres de la cadena resultante
136 \param tc pointer to a \ref bufr_tablec struct
137 \param index element to read if is not 0
138 \param d pointer to the source descriptor
139 \param ival integer value for the descriptor
140
141 If something went wrong, it returns NULL . Otherwise it returns \a expl
142*/
143char * bufrdeco_explained_table_val ( char *expl, size_t dim, struct bufr_tablec *tc, size_t *index, struct bufr_descriptor *d, uint32_t ival )
144{
145 char *c;
146 uint32_t nv, v, nl;
147 size_t i, j;
148
149 if ( tc->wmo_table )
150 {
151 // WMO case
152 return bufrdeco_explained_table_csv_val ( expl, dim, tc, index, d, ival );
153 }
154
155 if ( *index == 0 )
156 {
157 // here the calling b item learn where to find table C line
158 if ( bufr_find_tablec_index ( index, tc, d->c ) )
159 {
160 //sprintf ( b->error, "bufrdeco_explained_table_val(): descriptor '%s' not found in table D\n", d->c );
161 return NULL; // descritor not found
162 }
163 }
164
165 i = *index;
166 // reads the amount of possible values
167 if ( tc->l[i][7] != ' ' )
168 {
169 nv = strtoul ( &tc->l[i][7], &c, 10 );
170 }
171 else
172 {
173 return NULL;
174 }
175
176 // read a value
177 for ( j = 0; j < nv && i < tc->nlines ; i++ )
178 {
179 if ( tc->l[i][12] != ' ' )
180 {
181 v = strtoul ( &tc->l[i][12], &c, 10 );
182 j++;
183 if ( v != ival )
184 {
185 continue;
186 }
187 break;
188 }
189 }
190
191 if ( j == nv || i == tc->nlines )
192 {
193 return NULL; // Value not found
194 }
195
196 // read how many lines for the descriptors
197 nl = strtoul ( &tc->l[i][21], &c, 10 );
198
199
200 // if match then we have finished the search
201 strcpy ( expl, &tc->l[i][24] );
202 if ( nl > 1 )
203 {
204 for ( nv = 1 ; nv < nl; nv++ )
205 if ( ( strlen ( expl ) + strlen ( &tc->l[i + nv][22] ) ) < dim )
206 {
207 strcat ( expl, &tc->l[i + nv][22] );
208 }
209 }
210
211 return expl;
212}
