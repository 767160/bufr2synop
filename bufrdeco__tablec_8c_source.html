27  \fn int bufr_read_tablec ( struct bufr_tablec *tc, char *error )
28  \brief Reads a file with table C content according with ECMWF format
29  \param tc pointer to a target struct \ref bufr_tablec
30  \param error string where to set error if any
31 
32  If succeded return 0, otherwise 1
33 */
34 int bufr_read_tablec ( struct bufr_tablec *tc, char *error )
35 {
36  char aux[8], *c;
37  size_t startx = 0;
38  FILE *t;
39  size_t i = 0;
40  char caux[256];
41 
42  if ( tc->path == NULL )
43  {
44  return 1;
45  }
46 
47  // If we've already readed this table.
48  if ( strcmp ( tc->path, tc->old_path ) == 0 )
49  {
50  return 0; // all done
51  }
52 
53  strcpy ( caux, tc->path );
54  memset ( tc, 0, sizeof ( struct bufr_tablec ) );
55  strcpy ( tc->path,caux );
56  if ( ( t = fopen ( tc->path, "r" ) ) == NULL )
57  {
58  sprintf ( error,"Unable to open table C file '%s'\n", tc->path );
59  return 1;
60  }
61 
62  while ( fgets ( tc->l[i], 96, t ) != NULL && i < BUFR_MAXLINES_TABLEC )
63  {
64  // supress the newline
65  if ( ( c = strrchr ( tc->l[i],'\n' ) ) != NULL )
66  {
67  *c = '\0';
68  }
69  if ( tc->l[i][1] != ' ' && tc->l[i][2] != ' ' )
70  {
71  aux[0] = tc->l[i][1];
72  aux[1] = tc->l[i][2];
73  aux[2] = '\0';
74  startx = strtoul ( aux, &c, 10 );
75  if ( tc->x_start[startx] == 0 )
76  {
77  tc->x_start[startx] = i; // marc the start
78  }
79  }
80  ( tc->num[startx] ) ++;
81  i++;
82  }
83  fclose ( t );
84  tc->nlines = i;
85  tc->wmo_table = 0;
86  strcpy ( tc->old_path, tc->path ); // store latest path
87  return 0;
88 }

92  \brief Find the index of a line in table C for a given key of a descriptor
93  \param index pointer where to set the result
94  \param tc pointer to a struct \ref bufr_tablec where all table data is stored
95  \param key string in the form FXXYYY which is the key of descriptor we want to find out
96 
97  If the descriptor has been found with success then returns 0, othewise returns 1
98 */
99 int bufr_find_tablec_index ( size_t *index, struct bufr_tablec *tc, const char *key )
100 {
101  size_t i, i0;
102  size_t x = 0;
103  char *c, aux[8];
104 
105  aux[0] = key[1];
106  aux[1] = key[2];
107  aux[2] = '\0';
108  x = strtoul ( aux, &c, 10 );
109  i0 = tc->x_start[x];
110  for ( i = i0 ; i < i0 + tc->num[x] ; i++ )
111  {
112  if ( tc->l[i][0] != key[0] ||
113  tc->l[i][1] != key[1] ||
114  tc->l[i][2] != key[2] ||
115  tc->l[i][3] != key[3] ||
116  tc->l[i][4] != key[4] ||
117  tc->l[i][5] != key[5] )
118  {
119  continue;
120  }
121  else
122  {
123  *index = i;
124  return 0;
125  }
126  }
127  return 1; // not found
128 }
