26/*!
27 \fn int bufr_read_tableb_csv ( struct bufr_tableb *tb, char *error )
28 \brief Read a Table B file from a WMO csv formatted file and set the result in a struct \ref bufr_tableb
29 \param tb pointer to the struct \ref bufr_tableb where to set the results
30 \param error string with the error explanation if any
31
32 Note that this function assumes that file is csv formatted as WMO table B
33 Return 0 if success, 1 otherwise
34*/
35int bufr_read_tableb_csv ( struct bufr_tableb *tb, char *error )
36{
37 char *c;
38 FILE *t;
39 size_t i = 0;
40 int nt;
41 uint32_t ix;
42 char l[CSV_MAXL];
43 char caux[512];
44 char *tk[16];
45 struct bufr_descriptor desc;
46
47 if ( tb->path[0] == 0 )
48 {
49 return 1;
50 }
51
52 // If we've already readed this table. We just regenerate the table with original values
53 if ( strcmp ( tb->path, tb->old_path ) == 0 )
54 {
55 //printf ("Reutilizo tablas\n");
56 for ( i = 0; i < tb->nlines ; i++ )
57 {
58 tb->item[i].scale = tb->item[i].scale_ori;
59 tb->item[i].reference = tb->item[i].reference_ori;
60 tb->item[i].nbits = tb->item[i].nbits_ori;
61 tb->item[i].changed = 0;
62 }
63 return 0; // all done
64 }
65
66 strcpy ( caux, tb->path );
67 memset ( tb, 0, sizeof ( struct bufr_tableb ) );
68 strcpy ( tb->path, caux );
69 if ( ( t = fopen ( tb->path, "r" ) ) == NULL )
70 {
71 sprintf ( error,"Unable to open table B file '%s'\n", tb->path );
72 return 1;
73 }
74
75 // read first line, it is ignored
76 fgets ( l, CSV_MAXL, t );
77
78 while ( fgets ( l, CSV_MAXL, t ) != NULL && i < BUFR_MAXLINES_TABLEB )
79 {
80 // Parse line
81 if ( parse_csv_line ( &nt, tk, l ) < 0 || nt != 7 )
82 {
83 sprintf ( error,"Error parsing csv line from table B file '%s'\n", tb->path );
84 return 1;
85 }
86
87 // First we build the descriptor
88 ix = strtoul ( tk[0], &c, 10 );
89 uint32_t_to_descriptor ( &desc, ix );
90 tb->item[i].changed = 0; // Original from table B
91 tb->item[i].x = desc.x; // x
92 tb->item[i].y = desc.y; // y
93 strcpy ( tb->item[i].key, desc.c ); // key
94 if ( tb->num[desc.x] == 0 )
95 {
96 tb->x_start[desc.x] = i; // marc the start
97 }
98 tb->y_ref[desc.x][desc.y] = i - tb->x_start[desc.x]; // marc the position from start of first x
99 ( tb->num[desc.x] )++;
100
101 // detailed name
102 if ( tk[2][0] )
103 {
104 sprintf ( caux, "%s %s", tk[1], tk[2] );
105 }
106 else
107 {
108 strcpy ( caux, tk[1] );
109 }
110
111 // Check about length
112 if ( strlen ( caux ) >= BUFR_TABLEB_NAME_LENGTH )
113 caux[BUFR_TABLEB_NAME_LENGTH - 1] = 0;
114 // and finally copy
115 strcpy ( tb->item[i].name, caux );
116
117 // unit
118 strcpy ( caux, tk[3] );
119 if ( strlen ( caux ) >= BUFR_TABLEB_UNIT_LENGTH )
120 caux[BUFR_TABLEB_UNIT_LENGTH - 1] = 0;
121 strcpy ( tb->item[i].unit, caux );
122
123 // escale
124 tb->item[i].scale_ori = strtol ( tk[4], &c, 10 );
125 tb->item[i].scale = tb->item[i].scale_ori;
126
127 // reference
128 tb->item[i].reference_ori = strtol ( tk[5], &c, 10 );
129 tb->item[i].reference = tb->item[i].reference_ori;
130
131 // bits
132 tb->item[i].nbits_ori = strtol ( tk[6], &c, 10 );
133 tb->item[i].nbits = tb->item[i].nbits_ori;
134
135 i++;
136 }
137 tb->x_start[0] = 0; // fix the start for x = 0
138 fclose ( t );
139 tb->nlines = i;
140 tb->wmo_table = 1;
141 strcpy ( tb->old_path, tb->path ); // store latest path
142 return 0;
143}
