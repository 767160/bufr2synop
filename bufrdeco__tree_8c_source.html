26/*!
27 \fn int get_unexpanded_descriptor_array_from_sec3 ( struct bufr_sequence *s, struct bufrdeco *b )
28 \brief Set to a struct \ref bufr_sequence an unexpanded descriptor array from sec3 in a BUFR report
29 \param s Pointer to the target struct \ref bufr_sequence
30 \param b Pointer to the base struct \ref bufrdeco
31
32 Here we get a struct \ref bufr_sequence from the data about descriptors stored in sec3 of a bufr report
33*/
34int get_unexpanded_descriptor_array_from_sec3 ( struct bufr_sequence *s, struct bufrdeco *b )
35{
36 size_t i;
37
38 // First we copy the array descritors in sec3 as level0
39 for ( i = 0; i < b->sec3.ndesc ; i++ )
40 {
41 memcpy ( & ( s->lseq[i] ), & ( b->sec3.unexpanded[i] ), sizeof ( struct bufr_descriptor ) );
42 }
43 s->ndesc = b->sec3.ndesc;
44 return 0;
45}

48/*!
49 \fn int bufrdeco_parse_tree_recursive ( struct bufrdeco *b, struct bufr_sequence *father, const char *key )
50 \brief Parse the descriptor tree in a recursive way
51 \param key string with descriptor in form 'FXXYYY'
52 \param father pointer to the father struct \ref bufr_sequence
53 \param b pointer to the base struct \ref bufrdeco
54
55 Returns 0 if success, 1 otherwise
56 */
57int bufrdeco_parse_tree_recursive ( struct bufrdeco *b, struct bufr_sequence *father, const char *key )
58{
59 size_t i, nl;
60 struct bufr_sequence *l;
61
62 if ( key == NULL )
63 {
64 memset ( b->tree, 0, sizeof ( struct bufrdeco_expanded_tree ) );
65 // case first layer
66 b->tree->nseq = 1;
67 l = & ( b->tree->seq[0] );
68 strcpy ( l->key, "000000" );
69 l->level = 0;
70 l->father = NULL; // This layer is God, it has not father
71 l->iseq = 0;
72 strcpy(l->name, "Main sequence from SEC3");
73 // here we get ndesc and lsec array
74 if ( get_unexpanded_descriptor_array_from_sec3 ( l, b ) )
75 {
76 return 1;
77 }
78 }
79 else
80 {
81 if ( b->tree->nseq < NMAXSEQ_DESCRIPTORS )
82 {
83 ( b->tree->nseq ) ++;
84 }
85 else
86 {
87 sprintf ( b->error,"bufr_parse_tree_deep(): Reached max number of bufr_sequence. "
88 "Use bigger NMAXSEQ_LAYER \n" );
89 return 1;
90 }
91 nl = b->tree->nseq;
92 l = & ( b->tree->seq[nl - 1] );
93 strcpy ( l->key, key );
94 l->level = father->level + 1;
95 l->father = father;
96 l->iseq = nl - 1;
97 //printf ("level=%lu ", l->level);
98 // here we get ndesc and lsec array from table d
99 if ( bufrdeco_tabled_get_descriptors_array ( l, b, key ) )
100 {
101 return 1; // something went wrong
102 }
103 }
104
105 // now we detect sons and go to parse them
106 for ( i = 0; i < l->ndesc ; i++ )
107 {
108 // we search for descriptors with f == 3
109 if ( l->lseq[i].f != 3 )
110 {
111 l->sons[i] = NULL;
112 continue;
113 }
114 l->sons[i] = & ( b->tree->seq[b->tree->nseq] );
115 // we then recursively parse the son
116 if ( bufrdeco_parse_tree_recursive ( b, l, l->lseq[i].c ) )
117 {
118 return 1;
119 }
120 }
121
122 // if we are here all gone well
123 return 0;
124}

126/*!
127 \fn int bufrdeco_parse_tree ( struct bufrdeco *b )
128 \brief Parse the tree of descriptors without expand the replicators
129 \param b Pointer to the source struct \ref bufrdeco
130
131 This is the user function to parse the descriptor structure of a BUFR report. This is the first task
132 we need to perform after read the bufr report with the aid of \ref bufrdeco_read_bufr function.
133
134 At the end we will get an array of structs \ref bufr_sequence defining the tree
135
136 A sequence layer is needed when parsing expanded descriptor sec3 and sec4
137
138 First bufr_sequence is the sequence of descriptors in sec3 after
139 byte 8. This is a bufr_sequence in level 0.
140
141 When a sequence descriptor is found in a layer, the sequence entries found in table D
142 form this descriptor is a son bufr_sequence. This son has then a father
143 and also can have one or more sons. The index level is incremented by one every step it
144 go into decendents.
145
146 And so we go in a recursive way up to the end.
147
148 If success return 0, if something went wrong return 1
149*/
150int bufrdeco_parse_tree ( struct bufrdeco *b )
151{
152 // here we start the parse
153 return bufrdeco_parse_tree_recursive ( b, NULL, NULL );
154}
