29  \param s Pointer to the target struct \ref bufr_sequence

32  Here we get a struct \ref bufr_sequence from the data about descriptors stored in sec3 of a bufr report

34 int get_unexpanded_descriptor_array_from_sec3 ( struct bufr_sequence *s, struct bufrdeco *b )
35 {
36  size_t i;
37 
38  // First we copy the array descritors in sec3 as level0
39  for ( i = 0; i < b->sec3.ndesc ; i++ )

48 /*!
49  \fn int bufrdeco_parse_tree_recursive ( struct bufrdeco *b, struct bufr_sequence *father, const char *key )
50  \brief Parse the descriptor tree in a recursive way
51  \param key string with descriptor in form 'FXXYYY'
52  \param father pointer to the father struct \ref bufr_sequence
53  \param b pointer to the base struct \ref bufrdeco
54 
55  Returns 0 if success, 1 otherwise
56  */
57 int bufrdeco_parse_tree_recursive ( struct bufrdeco *b, struct bufr_sequence *father, const char *key )
58 {
59  size_t i, nl;
60  struct bufr_sequence *l;
61 
62  if ( key == NULL )
63  {
64  memset ( b->tree, 0, sizeof ( struct bufrdeco_expanded_tree ) );
65  // case first layer
66  b->tree->nseq = 1;
67  l = & ( b->tree->seq[0] );
68  strcpy ( l->key, "000000" );
69  l->level = 0;
70  l->father = NULL; // This layer is God, it has not father
71  // here we get ndesc and lsec array
72  if ( get_unexpanded_descriptor_array_from_sec3 ( l, b ) )
73  {
74  return 1;
75  }
76  }
77  else
78  {
79  if ( b->tree->nseq < NMAXSEQ_DESCRIPTORS )
80  {
81  ( b->tree->nseq ) ++;
82  }
83  else
84  {
85  sprintf ( b->error,"bufr_parse_tree_deep(): Reached max number of bufr_sequence. "
86  "Use bigger NMAXSEQ_LAYER \n" );
87  return 1;
88  }
89  nl = b->tree->nseq;
90  l = & ( b->tree->seq[nl - 1] );
91  strcpy ( l->key, key );
92  l->level = father->level + 1;
93  l->father = father;
94  //printf ("level=%lu ", l->level);
95  // here we get ndesc and lsec array from table d
96  if ( bufrdeco_tabled_get_descriptors_array ( l, b, key ) )
97  {
98  return 1; // something went wrong
99  }
100  }
101 
102  // now we detect sons and go to parse them
103  for ( i = 0; i < l->ndesc ; i++ )
104  {
105  // we search for descriptors with f == 3
106  if ( l->lseq[i].f != 3 )
107  {
108  l->sons[i] = NULL;
109  continue;
110  }
111  l->sons[i] = & ( b->tree->seq[b->tree->nseq] );
112  // we then recursively parse the son
113  if ( bufrdeco_parse_tree_recursive ( b, l, l->lseq[i].c ) )
114  {
115  return 1;
116  }
117  }
118 
119  // if we are here all gone well
120  return 0;
121 }

123 /*!
124  \fn int bufrdeco_parse_tree ( struct bufrdeco *b )
125  \brief Parse the tree of descriptors without expand the replicators
126  \param b Pointer to the source struct \ref bufrdeco
127 
128  This is the user function to parse the descriptor structure of a BUFR report. This is the first task
129  we need to perform after read the bufr report with the aid of \ref bufrdeco_read_bufr function.
130 
131  At the end we will get an array of structs \ref bufr_sequence defining the tree
132 
133  A sequence layer is needed when parsing expanded descriptor sec3 and sec4
134 
135  First bufr_sequence is the sequence of descriptors in sec3 after
136  byte 8. This is a bufr_sequence in level 0.
137 
138  When a sequence descriptor is found in a layer, the sequence entries found in table D
139  form this descriptor is a son bufr_sequence. This son has then a father
140  and also can have one or more sons. The index level is incremented by one every step it
141  go into decendents.
142 
143  And so we go in a recursive way up to the end.
144 
145  If success return 0, if something went wrong return 1
146 */
147 int bufrdeco_parse_tree ( struct bufrdeco *b )
148 {
149  // here we start the parse
150  return bufrdeco_parse_tree_recursive ( b, NULL, NULL );
151 }
