26/*!
27 \fn int bufr_read_tabled ( struct bufr_tabled *td, char *error )
28 \brief Reads a file with table D content according with ECMWF format
29 \param td pointer to a target struct \ref bufr_tabled
30 \param error string where to set error if any
31
32 If succeded return 0, otherwise 1
33*/
34int bufr_read_tabled ( struct bufr_tabled *td, char *error )
35{
36 char aux[8], *c;
37 FILE *t;
38 size_t ix = 0;
39 size_t i = 0;
40 char caux[256];
41
42 if ( td->path[0] == '\0' )
43 {
44 return 1;
45 }
46
47 // Check if we've already readed this table.
48 if ( strcmp ( td->path, td->old_path ) == 0 )
49 {
50 return 0; // all done
51 }
52
53 strcpy ( caux, td->path );
54 memset ( td, 0, sizeof ( struct bufr_tabled ) );
55 strcpy ( td->path,caux );
56 if ( ( t = fopen ( td->path, "r" ) ) == NULL )
57 {
58 sprintf ( error,"Unable to open table D file '%s'\n", td->path );
59 return 1;
60 }
61
62 while ( fgets ( td->l[i], 124, t ) != NULL && i < BUFR_MAXLINES_TABLED )
63 {
64 // supress the newline
65 if ( ( c = strrchr ( td->l[i],'\n' ) ) != NULL )
66 {
67 *c = '\0';
68 }
69 if ( td->l[i][1] != ' ' && td->l[i][2] != ' ' )
70 {
71 aux[0] = td->l[i][2];
72 aux[1] = td->l[i][3];
73 aux[2] = '\0';
74 ix = strtoul ( aux, &c, 10 );
75 if ( td->x_start[ix] == 0 )
76 {
77 td->x_start[ix] = i; // marc the start
78 }
79 }
80 ( td->num[ix] ) ++;
81 i++;
82 }
83 fclose ( t );
84 td->nlines = i;
85 td->wmo_table = 0;
86 strcpy ( td->old_path, td->path ); // store latest path
87 return 0;
88}

90/*!
91 \fn int bufr_find_tabled_index ( size_t *index, struct bufr_tabled *td, const char *key )
92 \brief Find the index of a line in table D for a given key of a descriptor
93 \param index pointer where to set the result
94 \param td pointer to a struct \ref bufr_tabled where all table data is stored
95 \param key string in the form FXXYYY which is the key of descriptor we want to find out
96
97 If the descriptor has been found with success then returns 0, othewise returns 1
98*/
99int bufr_find_tabled_index ( size_t *index, struct bufr_tabled *td, const char *key )
100{
101 size_t i, i0;
102 size_t ix = 0;
103 char *c, aux[8];
104
105 aux[0] = key[1];
106 aux[1] = key[2];
107 aux[2] = '\0';
108 ix = strtoul ( aux, &c, 10 );
109 i0 = td->x_start[ix];
110 for ( i = i0 ; i < i0 + td->num[ix] ; i++ )
111 {
112 if ( td->l[i][1] != key[0] ||
113 td->l[i][2] != key[1] ||
114 td->l[i][3] != key[2] ||
115 td->l[i][4] != key[3] ||
116 td->l[i][5] != key[4] ||
117 td->l[i][6] != key[5] )
118 {
119 continue;
120 }
121 else
122 {
123 *index = i;
124 return 0;
125 }
126 }
127 return 1; // not found
128}

131/*!
132 \fn int bufrdeco_tabled_get_descriptors_array ( struct bufr_sequence *s, struct bufrdeco *b, const char *key )
133 \brief get the descriptors array for a descriptor sequence defined in table D with F = 3
134 \param s target struct \ref bufr_sequence
135 \param b pointer to the basic container struct \ref bufrdeco
136 \param key string in the form FXXYYY which is the key of descriptor we want to find out, F = 3
137
138 If the sequence has been filled with success then returns 0, otherwise returns 1
139*/
140int bufrdeco_tabled_get_descriptors_array ( struct bufr_sequence *s, struct bufrdeco *b, const char *key )
141{
142 size_t i, j;
143 uint32_t nv, v;
144 char *c;
145 struct bufr_tabled *td;
146
147 td = & ( b->tables->d );
148
149 // Reject wrong arguments
150 if ( s == NULL || b == NULL || key == NULL )
151 {
152 sprintf ( b->error,"bufrdeco_tabled_get_descritors_array(): Wrong entry arguments\n" );
153 return 1;
154 }
155
156 // here the calling b item learn where to find table C line
157
158 if ( bufr_find_tabled_index ( &i, td, key ) )
159 {
160 sprintf ( b->error, "bufrdeco_tabled_val(): descriptor '%s' not found in table D\n", key );
161 return 1; // descritor not found
162 }
163
164 // Get the name of common sequence
165 if (td->item[i].description[0])
166 strcpy(s->name, td->item[i].description);
167 else
168 s->name[0] = 0;
169
170 // reads the amount of possible values
171 nv = strtoul ( &td->l[i][7], &c, 10 );
172
173 // s->level must be set by caller
174 // s->father must be set by caller
175
176 s->ndesc = 0;
177
178 // read all descriptors
179 for ( j = 0; j < nv && ( i + j ) < td->nlines ; j++ )
180 {
181 v = strtoul ( & ( td->l[i+j][11] ), &c, 10 );
182 uint32_t_to_descriptor ( & ( s->lseq[j] ), v );
183 ( s->ndesc ) ++;
184 }
185 // s->sons are not set here
186 return 0;
187}
