44 /*!
45  \fn int find_descriptor(int *haystack, size_t nlst, int needlemin, int needlemax)
46  \brief Try to find a descriptor in an array between two limits
47  \param haystack array of integers as descriptors
48  \param nlst number of descriptors in array
49  \param needlemin minimum descriptor to find
50  \param needlemax maximum descriptor to find
51 
52  It tries to find a descriptor in haystack which is greater or equal to needlemin and
53  lesser or equal to needlemax
54 
55  it returns 1 if found, 0 otherwise.
56 */
57 int find_descriptor_interval(int *haystack, size_t nlst, int needlemin, int needlemax)
58 {
59  size_t i = 0;
60  while ((haystack[i] > needlemax || haystack[i] < needlemin) && i < nlst)
61  i++;
62  return (i < nlst);
