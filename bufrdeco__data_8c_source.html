48 /*!
49  \fn int bufrdeco_decode_data_subset ( struct bufrdeco_subset_sequence_data *s, struct bufrdeco_compressed_data_references *r, struct bufrdeco *b )
50  \brief User interface to decode a BUFR subset
51  \param s pointer to the target struct \ref bufrdeco_subset_sequence_data
52  \param r pointer to the struct \ref bufrdeco_compressed_data_references
53  \param b pointer to the base struct \ref bufrdeco
54 
55  Note that if succeeded the counter to current subset index is increased. Remember that in case of non
56  compressed data we must to decode all previous subsets to get the desired one. In case of compressed
57  we can access directly to a the desired subset
58 
59  Return 0 in case of success, 1 otherwise
60 */
61 int bufrdeco_decode_data_subset ( struct bufrdeco_subset_sequence_data *s, struct bufrdeco_compressed_data_references *r,
62  struct bufrdeco *b )
63 {
64  // Check about parsed tree
65  if ( b->tree == NULL || b->tree->nseq == 0 )
66  {
67  sprintf ( b->error, "bufrdeco_subset_sequence_datad(): Try to parse compressed data without parsed tree\n" );
68  return 1;
69  }
70 
71  // First we init the subset sequence data
72  if ( bufrdeco_clean_subset_sequence_data ( s ) )
73  {
74  return 1;
75  }
76 
77  // Then we get the data from an already parsed descriptor tree
78  if ( b->sec3.compressed )
79  {
80  if ( r->nd == 0 /*r->refs == NULL || r->dim == 0 */ )
81  {
82  // case of compressed data and still not parsed
83  if ( bufrdeco_parse_compressed ( r, b ) )
84  {
85  return 1;
86  }
87  }
88  if ( bufr_decode_subset_data_compressed ( s, r, b ) )
89  {
90  return 1;
91  }
92  }
93  else
94  {
95  if ( bufrdeco_decode_subset_data_recursive ( s, NULL, b ) )
96  {
97  return 1;
98  }
99  }
100  // Finally we update the subset counter
101  ( b->state.subset ) ++;
102  return 0;
103 }

106 /*!
107  \fn int bufrdeco_increase_data_array ( struct bufrdeco_subset_sequence_data *s )
108  \brief doubles the allocated space for a struct \ref bufrdeco_subset_sequence_data whenever is posible
109  \param s pointer to source struct \ref bufrdeco_subset_sequence_data
110 
111  The amount of data in a bufr must be huge. In a first moment, the dimension of a sequence of structs
112  \ref bufr_atom_data is \ref BUFR_NMAXSEQ but may be increased. This function task is try to double the
113  allocated dimension and reallocate it.
114 
115  Return 0 when success, otherwise return 1 and the struct is unmodified
116 */
117 int bufrdeco_increase_data_array ( struct bufrdeco_subset_sequence_data *s )
118 {
119  if ( s->dim < ( BUFR_NMAXSEQ * 8 ) ) // check if reached the limit
120  {
121  if ( ( s->sequence = ( struct bufr_atom_data * ) realloc ( ( void * ) s->sequence,
122  s->dim * 2 * sizeof ( struct bufr_atom_data ) ) ) == NULL )
123  {
124  return 1;
125  }
126  else
127  {
128  s->dim *= 2;
129  return 0;
130  }
131  }
132  else
133  {
134  return 1;
135  }
136 }

138 /*!
139  \fn int bufrdeco_decode_subset_data_recursive ( struct bufrdeco_subset_sequence_data *s, struct bufr_sequence *l, struct bufrdeco *b )
140  \brief decode the data from a subset in a recursive way
141  \param s pointer to the target struct \ref bufrdeco_subset_sequence_data
142  \param l pointer to the source struct \ref bufr_sequence
143  \param b pointer to the base struct \ref bufrdeco
144 
145  Return 0 in case of success, 1 otherwise
146 */
147 int bufrdeco_decode_subset_data_recursive ( struct bufrdeco_subset_sequence_data *s, struct bufr_sequence *l, struct bufrdeco *b )
148 {
149  size_t i;
150  struct bufr_sequence *seq;
151  struct bufr_replicator replicator;
152 
153  // clean subset data
154  if ( l == NULL )
155  {
156  //memset ( s, 0, sizeof ( struct bufr_subset_sequence_data ) );
157  s->nd = 0;
158  seq = & ( b->tree->seq[0] );
159  if ( b->state.subset == 0 )
160  {
161  b->state.bit_offset = 0;
162  }
163  // also reset reference and bits inc
164  b->state.added_bit_length = 0;
165  b->state.added_scale = 0;
166  b->state.added_reference = 0;
167  b->state.assoc_bits = 0;
168  b->state.changing_reference = 255;
169  b->state.fixed_ccitt = 0;
170  b->state.local_bit_reserved = 0;
171  b->state.factor_reference = 1;
172  }
173  else
174  {
175  seq = l;
176  }
177 
178  // loop for a sequence
179  for ( i = 0; i < seq->ndesc ; i++ )
180  {
181  switch ( seq->lseq[i].f )
182  {
183  case 0:
184  // Get data from table B
185  if ( bufrdeco_tableb_val ( & ( s->sequence[s->nd] ), b, & ( seq->lseq[i] ) ) )
186  {
187  return 1;
188  }
189 
190  //bufr_print_atom_data_stdout(& ( s->sequence[s->nd] ));
191  if ( s->nd < ( s->dim - 1 ) )
192  {
193  ( s->nd ) ++;
194  }
195  else if ( bufrdeco_increase_data_array ( s ) == 0 )
196  {
197  ( s->nd ) ++;
198  }
199  else
200  {
201  sprintf ( b->error, "bufr_decode_data_subset_recursive(): No more bufr_atom_data available. Check BUFR_NMAXSEQ\n" );
202  return 1;
203  }
204  break;
205 
206  case 1:
207  // Case of replicator descriptor
208  replicator.s = seq;
209  replicator.ixrep = i;
210  if ( seq->lseq[i].y != 0 )
211  {
212  // no delayed
213  replicator.ixdel = i;
214  replicator.ndesc = seq->lseq[i].x;
215  replicator.nloops = seq->lseq[i].y;
216  bufrdeco_decode_replicated_subsequence ( s, &replicator, b );
217  i += replicator.ndesc;
218  }
219  else
220  {
221  // case of delayed;
222  replicator.ixdel = i + 1;
223  replicator.ndesc = seq->lseq[i].x;
224  // here we read ndesc from delayed replicator descriptor
225  if ( bufrdeco_tableb_val ( & ( s->sequence[s->nd] ), b, & ( seq->lseq[i + 1] ) ) )
226  {
227  return 1;
228  }
229  replicator.nloops = ( size_t ) ( s->sequence[s->nd].val );
230  if ( s->nd < ( s->dim - 1 ) )
231  {
232  ( s->nd ) ++;
233  }
234  else if ( bufrdeco_increase_data_array ( s ) == 0 )
235  {
236  ( s->nd ) ++;
237  }
238  else
239  {
240 
241  sprintf ( b->error, "bufr_decode_data_subset_recursive(): No more bufr_atom_data available. Check BUFR_NMAXSEQ\n" );
242  return 1;
243  }
244  bufrdeco_decode_replicated_subsequence ( s, &replicator, b );
245  i += replicator.ndesc + 1;
246  }
247  //i = replicator.ixdel + replicator.ndesc; // update i properly
248  break;
249 
250  case 2:
251  // Case of operator descriptor
252  if ( bufrdeco_parse_f2_descriptor ( s, & ( seq->lseq[i] ), b ) )
253  {
254  return 1;
255  }
256  break;
257 
258  case 3:
259  // Case of sequence descriptor
260  if ( bufrdeco_decode_subset_data_recursive ( s, seq->sons[i], b ) )
261  {
262  return 1;
263  }
264  break;
265 
266  default:
267  // this case is not possible
268  sprintf ( b->error, "bufr_decode_data_subset_recursive(): Found bad 'f' in descriptor\n" );
269  return 1;
270  break;
271  }
272  }
273  return 0;
274 };
