374 /*!
375  \fn int parse_temp_raw_data ( struct temp_chunks *t, struct temp_raw_data *r )
376  \brief parse a struct \ref temp_raw_data to fill chunks in a struct \ref temp_chunks
377 */
378 int parse_temp_raw_data ( struct temp_chunks *t, struct temp_raw_data *r )
379 {
380  int ix, is_over_100;
381  size_t i, j, isa = 0, isc = 0, ita = 0, itc = 0; // level counters
382  size_t iwxa = 0, iwxc = 0, itb = 0, itd = 0;
383  size_t iwd = 0, iwb = 0;

385 
386  if ( t == NULL || r == NULL )
387  return 1;
388 
389  if ( r->n == 0 )
390  return 1;
391 
392  // Some default
393  t->a.s1.id[0] = '/';
394  t->c.s1.id[0] = '/';
395  for ( i = 0; i < r->n; i++ )
396  {
397  d = & ( r->raw[i] ); // to make code easy
398 
399  if ( d->p < 10000.0 ) // to select which part
400  is_over_100 = 1;
401  else
402  is_over_100 = 0;
403 
404  // Surface data
405  if ( d->flags & TEMP_POINT_MASK_SURFACE )
406  {
407  pascal_to_pnpnpn ( t->a.s2.lev0.PnPnPn, d->p ); //PoPoPo
408  kelvin_to_TTTa ( t->a.s2.lev0.TnTnTan, d->T ); // TnTnTan
409  dewpoint_depression_to_DnDn ( t->a.s2.lev0.DnDn, d->T , d->Td ); // DnDn
410  wind_to_dndnfnfnfn ( t->a.s2.lev0.dndnfnfnfn, d->dd, d->ff ); // dndnfnfnfn
411  }
412 
413  // standard level
414  if ( d->flags & TEMP_POINT_MASK_STANDARD_LEVEL )
415  {
416  ix = ( int ) ( d->p * 0.01 + 0.5 ); // hPa
417  ix = ( ix / 10 ) % 100; // coded
418  if ( is_over_100 )
419  sprintf ( t->c.s2.std[isc].PnPn, "%02d", ix ); // PnPn
420  else
421  sprintf ( t->a.s2.std[isa].PnPn, "%02d", ix ); // PnPn
422 
423  ix = ( int ) ( d->h + 0.5 );
424  if ( d->p <= 50000.0 )
425  {
426  if ( is_over_100 )
427  sprintf ( t->c.s2.std[isc].hnhnhn, "%03d", ( ( ix + 5 ) / 10 ) % 1000 );
428  else
429  sprintf ( t->a.s2.std[isa].hnhnhn, "%03d", ( ( ix + 5 ) / 10 ) % 1000 );
430  }
431  else
432  {
433  if ( is_over_100 )
434  sprintf ( t->c.s2.std[isc].hnhnhn, "%03d", ix % 1000 );
435  else if ( ix >= 0 )
436  sprintf ( t->a.s2.std[isa].hnhnhn, "%03d", ix % 1000 );
437  else
438  sprintf ( t->a.s2.std[isa].hnhnhn, "%03d", ( -ix + 500 ) % 1000 );
439  }
440  if ( is_over_100 )
441  {
442  kelvin_to_TTTa ( t->c.s2.std[isc].TnTnTan, d->T ); // TnTnTan
443  dewpoint_depression_to_DnDn ( t->c.s2.std[isc].DnDn, d->T, d->Td ); // DnDn
444  wind_to_dndnfnfnfn ( t->c.s2.std[isc].dndnfnfnfn, d->dd, d->ff ); // dndnfnfnfn
445  // Now update Id
446  if ( d->ff != MISSING_REAL && d->dd != MISSING_REAL )
447  {

453  t->c.s2.n = isc;
454  }
455  }
456  else
457  {
458  kelvin_to_TTTa ( t->a.s2.std[isa].TnTnTan, d->T ); // TnTnTan
459  dewpoint_depression_to_DnDn ( t->a.s2.std[isa].DnDn, d->T , d->Td ); // DnDn
460  wind_to_dndnfnfnfn ( t->a.s2.std[isa].dndnfnfnfn, d->dd, d->ff ); // dndnfnfnfn
461  // Now update Id
462  if ( d->ff != MISSING_REAL && d->dd != MISSING_REAL )

464  t->a.s1.id[0] = t->a.s2.std[isa].PnPn[0];

509  {
510  ix = ( int ) ( d->p * 0.1 + 0.5 );

531  sprintf ( t->a.s4.windx[iwxa].PmPmPm, "%03d", ix % 1000 ); // PnPnPn.
532  wind_to_dndnfnfnfn ( t->a.s4.windx[iwxa].dmdmfmfmfm, d->dd, d->ff ); // dndnfnfnfn
533  for ( j = i + 1; j < r->n ; j++ )
