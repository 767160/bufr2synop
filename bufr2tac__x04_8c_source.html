26 /*!
27  \fn int time_period_duration (struct bufr2tac_subset_state *s )
28  \brief Get time period duration in seconds
29  \param s pinter to struct \ref bufr2tac_subset_state
30 
31  Returns -1 if no duration is computed. Otherwise returns time duration in seconds
32 */
33 int time_period_duration ( struct bufr2tac_subset_state *s )
34 {
35  if ( s->k_itval == ( s->k_jtval + 1 ) )
36  {
37  // two consecutive time period displacements
38  if ( ( s->itmask & DESCRIPTOR_VALUE_MISSING ) == 0 &&
39  ( s->jtmask & DESCRIPTOR_VALUE_MISSING ) == 0 )
40  {
41  return ( abs ( s->itval - s->jtval ) );
42  }
43  else if ( ( s->itmask & DESCRIPTOR_VALUE_MISSING ) &&
44  ( s->jtmask & DESCRIPTOR_VALUE_MISSING ) )
45  {
46  return 0;
47  }
48  else if ( ( s->itmask & DESCRIPTOR_VALUE_MISSING ) == 0 &&
49  ( s->jtmask & DESCRIPTOR_VALUE_MISSING ) )
50  {
51  return ( abs ( s->itval ) );
52  }
53  else if ( ( s->jtmask & DESCRIPTOR_VALUE_MISSING ) == 0 &&
54  ( s->itmask & DESCRIPTOR_VALUE_MISSING ) )
55  {
56  return -1;
57  }
58  }
59 
60  // just a single time period displacement
61  if ( (s->itmask & DESCRIPTOR_VALUE_MISSING ) == 0)
62  {
63  return ( abs ( s->itval ) );
64  }
65  else
66  {
67  return 0;
68  }
69 }

71 /*!
72  \fn int hour_rounded(struct synop_chunks *syn)
73  \brief Get the rounded hour of a given date
74  \param syn pointer to the \ref synop_chunks struct
75 
76  It returns the rounded hour if >= 0. If < 0 problems
77 */
78 int hour_rounded(struct synop_chunks *syn)
79 {
80  time_t t;
81  struct tm tim;
82  char aux[32];
83 
84  sprintf(aux, "%s%s%s%s%s", syn->e.YYYY,syn->e.MM,syn->e.DD,syn->e.HH, syn->e.mm);
85  memset(&tim, 0, sizeof(struct tm));
86  if (strlen(aux) == 12)
87  {
88  strptime(aux, "%Y%m%d%H%M", &tim);
89  t = mktime(&tim);
90  t += 1800;
91  gmtime_r(&t, &tim);
92  return tim.tm_hour;
93  }
94  else if (syn->e.mm[0] == 0 && syn->e.HH[0] )
95  {
96  return atoi(syn->e.HH);
97  }
98  else
99  return -1;
100 }

364 /*!

384  {
385  return 0;
386  }
387  if ( s->i && ( s->a1->desc.x == 4 ) && ( s->a1->desc.y == 1 ) )
388  {
389  // this is the final year of a normal period
390  s->is_normal = 1;
391  sprintf ( c->s2.YcYc, "%02d", s->ival % 100 );
