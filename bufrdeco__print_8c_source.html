102 /*!
103  \fn int bufrdeco_print_tree_recursive ( struct bufrdeco *b, struct bufr_sequence *seq )
104  \brief Print a tree of descriptors in a recursive way
105  \param b pointer to the basic container struct \ref bufrdeco
106  \param seq pointer to the struct \ref bufr_sequence to print
107 
108  If succeded return 0, else return 1
109 */
110 int bufrdeco_print_tree_recursive ( struct bufrdeco *b, struct bufr_sequence *seq )
111 {
112  size_t i, j;
113  struct bufr_sequence *l;
114 
115  if ( seq == NULL )
116  {
117  l = & ( b->tree->seq[0] );
118  }
119  else
120  {
121  l = seq;
122  }
123 
124  for ( i = 0; i < l->ndesc; i++ )
125  {
126  // we search for descriptors woth f == 3
127 
128  for ( j = 0; j < l->level; j++ )
129  {
130  printf ( " " );
131  }
132  printf ( "%u %02u %03u\n", l->lseq[i].f, l->lseq[i].x,l->lseq[i].y );
133 
134  if ( l->lseq[i].f != 3 )
135  {
136  continue;
137  }
138 
139  // we then recursively parse the son
140  if ( bufrdeco_print_tree_recursive ( b, l->sons[i] ) )
141  {
142  return 1;
143  }
144 
145  }
146 
147  return 0;
148 }

172 /*!
173  \fn char * bufrdeco_print_atom_data ( char *target, struct bufr_atom_data *a )
174  \brief print the data in a struct \ref bufr_atom_data to a string
175  \param target string where to print the result
176  \param a pointer to struct ref \ref bufr_atom_data with data to print
177 
178  Returns a pointer to result string
179 */
180 char * bufrdeco_print_atom_data ( char *target, struct bufr_atom_data *a )
181 {
182  char aux[256], *c;
183  c = target;
184  c += sprintf ( c, "%u %02u %03u ", a->desc.f, a->desc.x, a->desc.y );
185  strcpy ( aux, a->name );
186  aux[40] = '\0';
187  c += sprintf ( c, "%-40s ", aux );
188  strcpy ( aux, a->unit );
189  aux[20] = '\0';
190  c += sprintf ( c, "%-20s", aux );
191  if ( a->mask & DESCRIPTOR_VALUE_MISSING )
192  {
193  c += sprintf ( c, "%17s", "MISSING" );
194  }
195  else
196  {
197  if ( a->mask & DESCRIPTOR_HAVE_STRING_VALUE )
198  {
199  strcpy ( aux, a->cval );
200  aux[56] = '\0';
201  c += sprintf ( c, " " );
202  c += sprintf ( c, "%s", aux );
203  }
204  else if ( a->mask & DESCRIPTOR_HAVE_CODE_TABLE_STRING
205  || strstr ( a->unit, "CODE TABLE" ) == a->unit )
206  {
207  strcpy ( aux, a->ctable );
208  aux[56] = '\0';
209  c += sprintf ( c, "%17u ", ( uint32_t ) a->val );
210  c += sprintf ( c, "%s", aux );
211  }
212  else if ( a->mask & DESCRIPTOR_HAVE_FLAG_TABLE_STRING )
213  {
214  strcpy ( aux, a->ctable );
215  aux[56] = '\0';
216  c += sprintf ( c, " 0x%08X ", ( uint32_t ) a->val );
217  c += sprintf ( c, "%s", aux );
218  }
219  else
220  {
221  c += sprintf ( c, "%17.10e ", a->val );
222  }
223  }
224  return target;
225 }
