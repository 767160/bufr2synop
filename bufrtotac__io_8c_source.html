55 int read_args ( int _argc, char * _argv[] )
56 {
57  int iopt;
58 
59  // Initial and default values
60  INPUTFILE[0] = '\0';
61  OUTPUTFILE[0] = '\0';
62  LISTOFFILES[0] = '\0';
63  BUFRTABLES_DIR[0] = '\0';
64  VERBOSE = 0;
65  SHOW_SEQUENCE = 0;
66  DEBUG = 0;
67  NFILES = 0;
68  XML = 0;
69  JSON = 0;
70  CSV= 0;
71  ECMWF = 0;
72 
73  /*
74  Read input options
75  */
76  while ( ( iopt = getopt ( _argc, _argv, "cDEhi:jI:o:st:vVx" ) ) !=-1 )
77  switch ( iopt )
78  {
79  case 'i':
80  if ( strlen ( optarg ) < 256 )
81  strcpy ( INPUTFILE, optarg );
82  break;
83  case 'j':
84  JSON = 1;
85  break;
86  case 'I':
87  if ( strlen ( optarg ) < 256 )
88  strcpy ( LISTOFFILES, optarg );
89  break;
90  case 'o':
91  if ( strlen ( optarg ) < 256 )
92  strcpy ( OUTPUTFILE, optarg );
93  break;
94  case 't':
95  if ( strlen ( optarg ) < 256 )
96  {
97  strcpy ( BUFRTABLES_DIR, optarg );
98  }
99  break;
100  case 'D':
101  DEBUG = 1;
102  break;
103  case 'E':
104  ECMWF = 1;
105  break;
106  case 'V':
107  VERBOSE = 1;
108  break;
109  case 'v':
110  printf ( "%s %s\n", SELF, PACKAGE_VERSION );
111  exit ( EXIT_SUCCESS );
112  break;
113  case 's':
114  SHOW_SEQUENCE = 1;
115  break;
116  case 'x':
117  XML = 1;
118  break;
119  case 'c':
120  CSV = 1;
121  break;
122  case 'h':
123  default:
124  print_usage();
125  exit ( EXIT_SUCCESS );
126  }
127 
128  if ( INPUTFILE[0] == 0 && LISTOFFILES[0] == 0 )
129  {
130  printf ( "read_args(): It is needed an input file. Use -i or -I option\n" );
131  print_usage();
132  return -1;
133  }
134  return 1;

139  struct bufrdeco *b, char *err )
140 {
141  size_t i;
142  int ksec1[40], res;

144  size_t nlst = ( size_t ) b->sec3.ndesc;
145 
146  // memory for kdtlst
147  if ( ( kdtlst = ( int * ) calloc ( 1, nlst * sizeof ( int ) ) ) == NULL )
148  {
149  sprintf ( err, "decobufr_parse_subset_sequence(): cannot alloc memory for kdtlst\n" );
150  return 1;
151  }
152 
153  // sets descriptor as integer according to ECMWF
154  for ( i = 0; i < nlst ; i++ )
155  {
156  descriptor_to_integer ( &kdtlst[i], &b->sec3.unexpanded[i] );
157  }
158 
159  // And now set only used ksec1 elements
160  ksec1[5] = b->sec1.category;
161  ksec1[6] = b->sec1.subcategory_local;
162 
163  // Finaly we call to bufr2syn library
164  memset ( m, 0, sizeof ( struct metreport ) );
165  m->h = &b->header;
166  res = parse_subset_sequence ( m, &b->seq, st, kdtlst, nlst, ksec1, err );
167  free ( ( void * ) kdtlst );
168  return res;

172  \fn char * get_bufrfile_path( char *filename, char *err)
173  \brief Get bufr file names to parse
174  \param filename string where the file is on output
175  \param err string where to set the error if any
176 
177  If success, it returns a pointer to filename. If there is no more files
178  to parse, or error returns NULL.
179 
180  This routine is also opens the file with file list and close it when no more
181  file available in list
182 
183  The content format in LISTOFFILES is compatible with 'ls -1'. Example assume we
184  want to parse all *.bufr files in current directory, we can do
185 
186  \code
187  ls -1 *.bufr > list_file && bufrtotac -I list_file
188  \endcode
189 */
190 char * get_bufrfile_path ( char *filename, char *err )
191 {
192  char aux[256], *c;
193  if ( LISTOFFILES[0] == 0 )
194  {
195  if ( NFILES == 0 )
196  {
197  if ( filename != INPUTFILE )
198  strcpy ( filename, INPUTFILE );
199  return filename;
200  }
201  else
202  return 0;
203  }
204  else if ( NFILES == 0 )
205  {
206  if ( ( FL = fopen ( LISTOFFILES,"r" ) ) == NULL )
207  {
208  sprintf ( err,"Cannot open '%s'", LISTOFFILES );
209  return NULL;
210  }
211  }
212  if ( fgets ( aux, 256, FL ) )
213  {
214  c = strrchr ( aux,'\n' );
215  if ( c )
216  *c = 0;
217  strcpy ( filename, aux );
218  return filename;
219  }
220  else
221  {
222  fclose ( FL );
223  return NULL;
224  }
225 }
