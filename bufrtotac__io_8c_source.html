56/*!
57 \fn char *bufrtotac_get_version(char *version, char *build, char *builder, int *version_major, int *version_minor, int *version_patch)
58 \brief Get strings with version information and build date and time
59 \param version pointer to string with version as result if not NULL
60 \param build pointer to string with compiler and compilation date and time if not NULL
61 \param builder pointer to string with builder utility. 'cmake' or 'autotools' if not NULL
62 \param version_major pointer to string with version_major component if not NULL
63 \param version_minor pointer to string with version_minor component if not NULL
64 \param version_patch pointer to string with version_patch component if not NULL
65
66 Retuns string pointer \ref version.
67 */
68char *bufrtotac_get_version(char *version, char *build, char *builder, int *version_major, int *version_minor, int *version_patch)
69{
70 int major = 0, minor = 0, patch = 0;
71 char *c;
72
73 if (version == NULL)
74 return NULL;
75 sprintf(version, "%s", VERSION);
76 // default values
77 sscanf(version, "%d.%d.%d", &major, &minor, &patch);
78
79 if (build)
80 {
81 c = build;
82#if defined(__INTEL_COMPILER)
83 c += sprintf(build, "using INTEL C compiler icc %d.%d ", __INTEL_COMPILER, __INTEL_COMPILER_UPDATE);
84#elif defined(__GNUC__)
85 c += sprintf(build, "using GNU C compiler gcc %d.%d.%d ", __GNUC__ , __GNUC_MINOR__ , __GNUC_PATCHLEVEL__);
86#endif
87 sprintf(c,"at %s %s",__DATE__,__TIME__);
88 }
89
90 if (builder)
91#ifdef BUILD_USING_CMAKE
92 strcpy(builder, "cmake");
93#else
94 strcpy(builder, "autotools");
95#endif
96 if (version_major)
97 *version_major = major;
98 if (version_minor)
99 *version_minor = minor;
100 if (version_patch)
101 *version_patch = patch;
102 return version;
103}

118/*!
119 \fn int read_args( int _argc, char * _argv[])
120 \brief read the arguments from stdio
121 \param _argc number of arguments passed
122 \param _argv array of arguments
123
124 Returns 1 if succcess, -1 othewise
125*/
126int read_args ( int _argc, char * _argv[] )
127{
128 int iopt;
129 char *c;
130 char aux[128];
131
132 // Initial and default values
133 INPUTFILE[0] = '\0';
134 OUTPUTFILE[0] = '\0';
135 LISTOFFILES[0] = '\0';
136 BUFRTABLES_DIR[0] = '\0';
137 VERBOSE = 0;
138 SHOW_SEQUENCE = 0;
139 DEBUG = 0;
140 NFILES = 0;
141 XML = 0;
142 JSON = 0;
143 CSV= 0;
144 EXTRACT = 0;
145 ECMWF = 0;
146 HTML = 0;
147 NOTAC = 0;
148 FIRST_SUBSET = 0;
149 LAST_SUBSET = BUFR_LEN;
150
151 /*
152 Read input options
153 */
154 while ( ( iopt = getopt ( _argc, _argv, "cDEhi:jHI:no:S:st:vVxX" ) ) !=-1 )
155 switch ( iopt )
156 {
157 case 'i':
158 if ( strlen ( optarg ) < 256 )
159 strcpy ( INPUTFILE, optarg );
160 break;
161 case 'j':
162 JSON = 1;
163 break;
164 case 'I':
165 if ( strlen ( optarg ) < 256 )
166 strcpy ( LISTOFFILES, optarg );
167 break;
168 case 'n':
169 NOTAC = 1;
170 break;
171 case 'o':
172 if ( strlen ( optarg ) < 256 )
173 strcpy ( OUTPUTFILE, optarg );
174 break;
175 case 't':
176 if ( strlen ( optarg ) < 256 )
177 {
178 strcpy ( BUFRTABLES_DIR, optarg );
179 }
180 break;
181 case 'D':
182 DEBUG = 1;
183 break;
184 case 'E':
185#ifdef USE_BUFRDC
186 ECMWF = 1;
187#endif
188 break;
189 case 'H':
190 HTML = 1;
191 break;
192 case 'V':
193 VERBOSE = 1;
194 break;
195 case 'v':
196 print_version();
197 exit ( EXIT_SUCCESS );
198 break;
199 case 's':
200 SHOW_SEQUENCE = 1;
201 break;
202 case 'S':
203 // Process subset sequence to show
204 if ( strlen ( optarg ) < 128 &&
205 strspn ( optarg, "0123456789." ) == strlen ( optarg ) )
206 {
207 if ( strstr ( optarg, ".." ) != NULL )
208 {
209 strcpy ( aux, optarg );
210 c = strstr ( aux, ".." );
211 if ( strlen ( c ) > 2 )
212 {
213 LAST_SUBSET = atoi ( c + 2 );
214 }
215 *c = '\0';
216 if ( aux[0] )
217 {
218 FIRST_SUBSET = atoi ( aux );
219 }
220 }
221 else
222 {
223 FIRST_SUBSET = atoi ( optarg );
224 LAST_SUBSET = FIRST_SUBSET;
225 }
226 }
227 break;
228 case 'x':
229 XML = 1;
230 break;
231 case 'c':
232 CSV = 1;
233 break;
234 case 'X':
235 EXTRACT = 1;
236 break;
237 case 'h':
238 default:
239 print_usage();
240 exit ( EXIT_SUCCESS );
241 }
242
243 if ( INPUTFILE[0] == 0 && LISTOFFILES[0] == 0 )
244 {
245 printf ( "read_args(): It is needed an input file. Use -i or -I option\n" );
246 print_usage();
247 return -1;
248 }
249 return 1;
250}

253int bufrdeco_parse_subset_sequence ( struct metreport *m, struct bufr2tac_subset_state *st,
254 struct bufrdeco *b, char *err )
255{
256 size_t i;
257 int ksec1[40], res;
258 int *kdtlst;
259 size_t nlst = ( size_t ) b->sec3.ndesc;
260
261 // memory for kdtlst
262 if ( ( kdtlst = ( int * ) calloc ( 1, nlst * sizeof ( int ) ) ) == NULL )
263 {
264 sprintf ( err, "decobufr_parse_subset_sequence(): cannot alloc memory for kdtlst\n" );
265 return 1;
266 }
267
268 // sets descriptor as integer according to ECMWF
269 for ( i = 0; i < nlst ; i++ )
270 {
271 descriptor_to_integer ( &kdtlst[i], &b->sec3.unexpanded[i] );
272 }
273
274 // And now set only used ksec1 elements
275 ksec1[5] = b->sec1.category;
276 ksec1[6] = b->sec1.subcategory_local;
277
278 // Finaly we call to bufr2tac library
279 memset ( m, 0, sizeof ( struct metreport ) );
280 m->h = &b->header;
281 res = parse_subset_sequence ( m, &b->seq, st, kdtlst, nlst, ksec1, err );
282 free ( ( void * ) kdtlst );
283 return res;
284}

286/*!
287 \fn char * get_bufrfile_path( char *filename, char *err)
288 \brief Get bufr file names to parse
289 \param filename string where the file is on output
290 \param err string where to set the error if any
291
292 If success, it returns a pointer to filename. If there is no more files
293 to parse, or error returns NULL.
294
295 This routine is also opens the file with file list and close it when no more
296 file available in list
297
298 The content format in LISTOFFILES is compatible with 'ls -1'. Example assume we
299 want to parse all *.bufr files in current directory, we can do
300
301 \code
302 ls -1 *.bufr > list_file && bufrtotac -I list_file
303 \endcode
304*/
305char * get_bufrfile_path ( char *filename, char *err )
306{
307 char aux[256], *c;
308 if ( LISTOFFILES[0] == 0 )
309 {
310 if ( NFILES == 0 )
311 {
312 if ( filename != INPUTFILE )
313 strcpy ( filename, INPUTFILE );
314 return filename;
315 }
316 else
317 return 0;
318 }
319 else if ( NFILES == 0 )
320 {
321 if ( ( FL = fopen ( LISTOFFILES,"r" ) ) == NULL )
322 {
323 sprintf ( err,"Cannot open '%s'", LISTOFFILES );
324 return NULL;
325 }
326 }
327 if ( fgets ( aux, 256, FL ) )
328 {
329 c = strrchr ( aux,'\n' );
330 if ( c )
331 *c = 0;
332 strcpy ( filename, aux );
333 return filename;
334 }
335 else
336 {
337 fclose ( FL );
338 return NULL;
339 }
340}
