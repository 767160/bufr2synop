73 /*!
74  \fn int parse_csv_line(int *nt, char *tk[], char *lin)
75  \brief Parse a csv line
76  \param nt pointer to a integer. On success is the number of items found
77  \param tk array of pointers. Every pointer is a item on success
78  \param lin input line which is modified in this routine to be splitted into items
79 
80  NOTE that input line is modified
81  On success return 0, otherwise -1
82 */
83 int parse_csv_line ( int *nt, char *tk[], char *lin )
84 {
85  size_t i, j, k = 0, l, latest_char = 0;
86  int flag;
87  char c, caux[CSV_MAXL];
88 
89  l = strlen ( lin );
90  if ( l > CSV_MAXL || l == 0 )
91  return -1;
92 
93  // copy original string
94  strcpy ( caux, lin );
95 
96  flag = CSV_WAIT_ITEM;
97 
98  for ( i = 0, j = 0, k = 0; i < l; i++ )
99  {
100  c = caux[i]; // original char
101 
102  if ( ( flag & CSV_IS_CITED ) == 0 )
103  {
104  if ( flag & CSV_WAIT_ITEM )
105  {
106  if ( strchr ( CSV_SPACE, c ) != NULL )
107  {
108  lin[j++] = c; // copy to target
109  continue;
110  }
111  else if ( c == CSV_CITE )
112  {
113  lin[j++] = '\0'; // copy to target
114  flag |= CSV_IS_CITED;
115  continue;
116  }
117  else if ( c == CSV_SEPARATOR )
118  {
119  /* item vacio */
120  tk[k++] = lin + j;
121  lin[j++] = '\0';
122  flag = CSV_WAIT_ITEM;
123  continue;
124  }
125  else if ( c == CSV_FINAL )
126  {
127  if ( flag & CSV_FINISHED_ITEM )
128  {
129  *nt = k;
130  return 0;
131  }
132  /* al menos un token, vacio */
133  tk[k++] = lin + j;
134  lin[j] = '\0';
135  *nt = k;
136  return 0;
137  }
138  else
139  {
140  latest_char = j;
141  tk[k] = lin + j;
142  lin[j++] = c;
143  flag = CSV_IS_ITEM;
144  continue;
145  }
146  }
147  else if ( flag & CSV_IS_ITEM )
148  {
149  if ( strchr ( CSV_SPACE, c ) != NULL )
150  {
151  lin[j++] = c;
152  continue;
153  }
154  else if ( c == CSV_SEPARATOR )
155  {
156  lin[latest_char + 1] = '\0';
157  j = latest_char + 2;
158  k++;
159  flag = CSV_WAIT_ITEM | CSV_FINISHED_ITEM;
160  continue;
161  }
162  else if ( c == CSV_FINAL )
163  {
164  lin[latest_char + 1] = '\0';
165  k++;
166  *nt = k;
167  return 0;
168  }
169  else
170  {
171  latest_char = j++;
172  }
173  }
174  else if ( flag & CSV_WAIT_SEPARATOR )
175  {
176  if ( c == CSV_SEPARATOR )
177  {
178  lin[j++] = '\0';
179  k++;
180  flag = CSV_WAIT_ITEM;
181  }
182  else if ( c == CSV_FINAL )
183  {
184  lin[j++] = '\0';
185  *nt = k + 1;
186  return 0;
187  }
188  continue;
189  }
190  }
191  else /* CITED */
192  {
193  if ( flag & CSV_WAIT_ITEM )
194  {
195  tk[k] = lin + j;
196  flag &= ( ~CSV_WAIT_ITEM ); // Clean wait item bit
197  }
198  if ( c == CSV_CITE )
199  {
200  if ( caux[i + 1] == CSV_CITE )
201  {
202  if ( flag & CSV_IS_DOUBLE_QUOTED )
203  {
204  flag &= ( ~CSV_IS_DOUBLE_QUOTED ) ; // Clean double quoted bit
205  lin[j++] = '"';
206  i++;
207  }
208  else
209  {
210  flag |= CSV_IS_DOUBLE_QUOTED;
211  lin[j++] = '"';
212  i++;
213  }
214  continue;
215  }
216  else
217  {
218  lin[j++] = '\0';
219  flag = CSV_WAIT_SEPARATOR;
220  }
221  }
222  else if ( c == CSV_FINAL )
223  {
224  lin[j++] = '\0';
225  *nt = k + 1;
226  return 0;
227  }
228  else
229  {
230  lin[j++] = c;
231  }
232  continue;
233  }
234  }
235  return -1;
236 }
