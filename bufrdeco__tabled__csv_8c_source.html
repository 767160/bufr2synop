26/*!
27 \fn int bufr_read_tabled_csv ( struct bufr_tabled *td, char *error )
28 \brief Reads a file with table D content according with WMO csv format
29 \param td pointer to a target struct \ref bufr_tabled
30 \param error string where to set error if any
31
32 If succeded return 0, otherwise 1
33*/
34int bufr_read_tabled_csv ( struct bufr_tabled *td, char *error )
35{
36 char aux[32], *c;
37 char *tk[16];
38 int nt;
39 FILE *t;
40 size_t ix = 0;
41 size_t i = 0;
42 int j0, nj = 0;
43 char oldkey[8];
44 char caux[256];
45 char laux[CSV_MAXL];
46 struct bufr_descriptor desc;
47
48 if ( td->path[0] == 0 )
49 {
50 return 1;
51 }
52
53 // Check if we've already readed this table.
54 if ( strcmp ( td->path, td->old_path ) == 0 )
55 {
56 return 0; // all done
57 }
58
59 strcpy ( caux, td->path );
60 memset ( td, 0, sizeof ( struct bufr_tabled ) );
61 strcpy ( td->path,caux );
62 if ( ( t = fopen ( td->path, "r" ) ) == NULL )
63 {
64 sprintf ( error,"Unable to open table D file '%s'\n", td->path );
65 return 1;
66 }
67
68 // read first line, it is ignored
69 fgets ( laux, CSV_MAXL, t );
70
71 oldkey[0] = 0;
72 j0 = 0;
73
74 while ( fgets ( laux, CSV_MAXL, t ) != NULL && i < BUFR_MAXLINES_TABLED )
75 {
76 // Parse line
77 //printf("%s\n",laux);
78 if ( parse_csv_line ( &nt, tk, laux ) < 0 || ( nt != 2 && nt != 4) )
79 {
80 sprintf ( error,"Error parsing csv line from table D file '%s' found %d tokens in line %lu\n", td->path, nt, i );
81 return 1;
82 }
83
84 // item fields
85 strcpy(td->item[i].key, tk[0]);
86 strcpy(td->item[i].key2, tk[1]);
87
88 if (nt == 4 && tk[2][0])
89 strcpy(td->item[i].description, tk[2]);
90 else
91 td->item[i].description[0] = 0;
92
93 if (nt == 4 && tk[3][0])
94 strcpy(td->item[i].description2, tk[3]);
95 else
96 td->item[i].description2[0] = 0;
97
98 if ( strcmp ( oldkey, tk[0] ) )
99 {
100 if ( oldkey[0] )
101 {
102 // write number of descriptors in emulated ECMWF line
103 sprintf ( aux, "%s%3d", oldkey, nj );
104 td->l[j0][1] = aux[0];
105 td->l[j0][2] = aux[1];
106 td->l[j0][3] = aux[2];
107 td->l[j0][4] = aux[3];
108 td->l[j0][5] = aux[4];
109 td->l[j0][6] = aux[5];
110 td->l[j0][7] = aux[6];
111 td->l[j0][8] = aux[7];
112 td->l[j0][9] = aux[8];
113 }
114 j0 = i;
115 nj = 1;
116 }
117 else
118 {
119 nj++;
120 }
121 strcpy ( oldkey, tk[0] );
122
123 ix = strtoul ( tk[0], &c, 10 );
124 uint32_t_to_descriptor ( &desc, ix );
125
126 if ( td->num[desc.x] == 0 )
127 {
128 td->x_start[desc.x] = i; // marc the start
129 }
130 ( td->num[desc.x] ) ++;
131
132 // Now emule ECMWF format
133 sprintf ( td->l[i], " %s", tk[1] );
134 i++;
135 }
136 fclose ( t );
137
138 // Last sequence
139 sprintf ( aux, "%s%3d", oldkey, nj );
140 td->l[j0][1] = aux[0];
141 td->l[j0][2] = aux[1];
142 td->l[j0][3] = aux[2];
143 td->l[j0][4] = aux[3];
144 td->l[j0][5] = aux[4];
145 td->l[j0][6] = aux[5];
146 td->l[j0][7] = aux[6];
147 td->l[j0][8] = aux[7];
148 td->l[j0][9] = aux[8];
149
150 /*for ( kk = 0; kk < i; kk++ )
151 printf ( "%s\n", td->l[kk] );*/
152 td->nlines = i;
153 td->wmo_table = 1;
154 strcpy ( td->old_path, td->path ); // store latest path
155 return 0;
156}
