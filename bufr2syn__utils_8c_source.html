83 /*! \fn int tokenize_string(char *tk[], size_t ntk, char *target, size_t len, char *blanks)
84  \brief Split the report string into tokens
85  \param tk array of strings. Memory have to be allocated previously by caller
86  \param ntk max amount of elements in tk array.
87  \param target string which is the report to split
88  \param len lenght of \a target string
89  \param blanks string with the character considered as blanks. If null
90  the default " =\n\r\v\t" is considered
91  Returns the number of tokens \a target has been splitted into.
92  Retuns 0 in case of a void or too large target.
93  NOTE: \a target is changed after calling this routine. In fact, the pointers
94  in \a tk[] are linking to some char in string \a target
95 */
96 size_t tokenize_string(char *tk[], size_t ntk, char *target, size_t len, char *blanks)
97 {
98  size_t i = 1, n;
99  char d[] = " =\n\r\v\t", *b, *aux;
100 
101  // set the blanks array
102  if (blanks == NULL)
103  b = &d[0];
104  else
105  b = blanks;
106 
107  // check target string limits
108  n = strlen(target);
109  if (n == 0 || n > len)
110  return 0;
111 
112  // the split loop
113  tk[0] = strtok_r(target, b, &aux);
114  while (i < ntk &&
115  ((tk[i] = strtok_r(NULL, b, &aux)) != NULL ) &&
116  strlen(tk[i]) > 2 &&
117  strlen(tk[i]) < 10)
118  i++;
119  if (i < ntk)
120  return i;
121  else
122  return ntk - 1;
123 }

305 int guess_gts_header(struct gts_header *h, const char *f)
306 {
307  size_t nt;
308  char aux[256], *tk[16], *c;
309 
310  // parse file name
311  strcpy(h->filename, f);
312  strcpy(aux, f);
313 
314  // check latest '/' in filename
315  if ((c = strrchr(aux,'/')) == NULL)
316  c = &aux[0];
317  else
318  c++;
319 
320  nt = tokenize_string(tk, 16, aux, strlen(c), "_.");
321  // parse filenames with format as example 'AAAAMMDDHHmmss_ISIE06_SBBR_012100_RRB.bufr'
322 
323  // 5 or 6 items
324  if ( nt < 5 || nt > 6)
325  return 0;
326 
327  // extension bufr
328  if (strcmp(tk[nt - 1],"bufr"))
329  return 0;
330 
331  // item 0 the timestamp of file in NOAA GTS gateway
332  if (strlen(tk[0]) != 14 || strspn(tk[0], "0123456789") != 14)
333  return 0;
334  strcpy(h->timestamp, tk[0]);
335 
336  // item 1
337  if (strlen(tk[1]) != 6 || strspn(&tk[1][4],"0123456789") != 2)
338  return 0;
339  strcpy(h->bname, tk[1]);
340 
341  // item 2
342  if (strlen(tk[2]) != 4)
343  return 0;
344  strcpy(h->center, tk[2]);
345 
346  if (strlen(tk[3]) != 6 || strspn(tk[3], "0123456789") != 6)
347  return 0;
348  strcpy(h->dtrel, tk[3]);
349 
350  if (nt == 5)
351  {
352  strcpy(h->order, "BBB");
353  }
354  else
355  {
356  if (strlen(tk[4]) == 3)
357  strcpy(h->order, tk[4]);
358  else
359  return 0;
360  }
361 
362  return 1;
363 }
