47/*!
48 \fn int read_args( int _argc, char * _argv[])
49 \brief read input arguments with the aid of getopt
50 */
51int read_args ( int _argc, char * _argv[] )
52{
53 int iopt;
54
55 // Initial and default values
56 INPUTFILE[0] = '\0';
57 OUTPUTFILE[0] = '\0';
58 LISTOFFILES[0] = '\0';
59 BUFRTABLES_DIR[0] = '\0';
60 VERBOSE = 0;
61 SHOW_SEQUENCE = 0;
62 SHOW_ECMWF_OUTPUT = 0;
63 DEBUG = 0;
64 NFILES = 0;
65 XML = 0;
66 JSON = 0;
67 CSV= 0;
68
69 /*
70 Read input arguments using getop library
71 */
72 while ( ( iopt = getopt ( _argc, _argv, "cDehi:jI:o:st:vVx" ) ) != -1 )
73 switch ( iopt )
74 {
75 case 'i':
76 if ( strlen ( optarg ) < 256 )
77 strcpy ( INPUTFILE, optarg );
78 break;
79 case 'j':
80 JSON = 1;
81 break;
82 case 'I':
83 if ( strlen ( optarg ) < 256 )
84 strcpy ( LISTOFFILES, optarg );
85 break;
86 case 'o':
87 if ( strlen ( optarg ) < 256 )
88 strcpy ( OUTPUTFILE, optarg );
89 break;
90 case 't':
91 if ( strlen ( optarg ) < 256 )
92 {
93 strcpy ( BUFRTABLES_DIR, optarg );
94 }
95 break;
96 case 'D':
97 DEBUG = 1;
98 break;
99 case 'e':
100 SHOW_ECMWF_OUTPUT = 1;
101 break;
102 case 'V':
103 VERBOSE = 1;
104 break;
105 case 'v':
106 printf ( "%s %s\n", SELF, PACKAGE_VERSION );
107 exit ( EXIT_SUCCESS );
108 break;
109 case 's':
110 SHOW_SEQUENCE = 1;
111 break;
112 case 'x':
113 XML = 1;
114 break;
115 case 'c':
116 CSV = 1;
117 break;
118 case 'h':
119 default:
120 print_usage();
121 exit ( EXIT_SUCCESS );
122 }
123
124 if ( INPUTFILE[0] == 0 && LISTOFFILES[0] == 0 )
125 {
126 print_usage();
127 return 1;
128 }
129 return 0;
130}

133/*!
134 \fn char * get_bufrfile_path( char *filename, char *err)
135 \brief Get bufr file names to parse
136 \param filename string where the file is on output
137 \param err string where to set the error if any
138
139 If success, it returns a pointer to filename. If there is no more files
140 to parse, or error returns NULL.
141
142 This routine is also opens the file with file list and close it when no more
143 file available in list
144
145 The content format in LISTOFFILES is compatible with 'ls -1'. Example assume we
146 want to parse all *.bufr files in current directory, we can do
147
148 \code
149 ls -1 *.bufr > list_file && bufr2synop -I list_file
150 \endcode
151*/
152char * get_bufrfile_path ( char *filename, char *err )
153{
154 char aux[256], *c;
155 if ( LISTOFFILES[0] == 0 )
156 {
157 if ( NFILES == 0 )
158 {
159 strcpy ( filename, INPUTFILE );
160 return filename;
161 }
162 else
163 return 0;
164 }
165 else if ( NFILES == 0 )
166 {
167 if ( ( FL = fopen ( LISTOFFILES,"r" ) ) == NULL )
168 {
169 sprintf ( err,"Cannot open '%s'", LISTOFFILES );
170 return NULL;
171 }
172 }
173 if ( fgets ( aux, 256, FL ) )
174 {
175 c = strrchr ( aux,'\n' );
176 if ( c )
177 *c = 0;
178 strcpy ( filename, aux );
179 return filename;
180 }
181 else
182 {
183 fclose ( FL );
184 return NULL;
185 }
186}
