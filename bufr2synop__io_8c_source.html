113  default:

127 /*!
128  \fn char * get_bufrfile_path( char *filename, char *err)
129  \brief Get bufr file names to parse
130  \param filename string where the file is on output
131  \param err string where to set the error if any
132 
133  If success, it returns a pointer to filename. If there is no more files
134  to parse, or error returns NULL.
135 
136  This routine is also opens the file with file list and close it when no more
137  file available in list
138 
139  The content format in LISTOFFILES is compatible with 'ls -1'. Example assume we
140  want to parse all *.bufr files in current directory, we can do
141 
142  \code
143  ls -1 *.bufr > list_file && bufr2synop -I list_file
144  \endcode
145 */
146 char * get_bufrfile_path ( char *filename, char *err )
147 {
148  char aux[256], *c;
149  if ( LISTOFFILES[0] == 0 )
150  {
151  if ( NFILES == 0 )
152  {
153  strcpy ( filename, INPUTFILE );
154  return filename;
155  }
156  else
157  return 0;
158  }
159  else if ( NFILES == 0 )
160  {
161  if ( ( FL = fopen ( LISTOFFILES,"r" ) ) == NULL )
162  {
163  sprintf ( err,"Cannot open '%s'", LISTOFFILES );
164  return NULL;
165  }
166  }
167  if ( fgets ( aux, 256, FL ) )
168  {
169  c = strrchr ( aux,'\n' );
170  if ( c )
171  *c = 0;
172  strcpy ( filename, aux );
173  return filename;
174  }
175  else
176  {
177  fclose ( FL );
178  return NULL;
179  }
180 }
