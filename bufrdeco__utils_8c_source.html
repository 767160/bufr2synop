31  \fn size_t get_bits_as_char_array ( char *target, uint8_t *has_data, uint8_t *source, size_t *bit0_offset, size_t bit_length )
32  \brief Get a string from an array of bits
33  \param target string as result
34  \param has_data Output flags to check whether is missing data. If 0 then data is missing, othewise has data
35  \param source array of uint8_t elements. Most significant bit of first element is the bit offest reference
36  \param bit0_offset Bit offset
37  \param bit_length Lenght (in bits) for the chunck to extract
38 
39  If returns the amount of bits readed. 0 if problems. It also update bits_offset with the new bits.

41 size_t get_bits_as_char_array2 ( char *target, uint8_t *has_data, uint8_t *source, size_t *bit0_offset, size_t bit_length )
42 {
43  size_t i, j;
44  size_t r, d, nc;
45  uint8_t *c;
46 
47  if ( bit_length % 8 )
48  return 0; // bit_length needs to be divisible by 8
49 
50  nc = bit_length / 8;
51  *has_data = 0; // marc if no missing data is present
52  for ( j = 0; j < nc ; j++ )
53  {
54  r = 8;
55  d = 0;
56  * ( target + j ) = 0;
57  do
58  {
59  c = source + ( *bit0_offset + d ) / 8;
60  i = ( *bit0_offset + d ) % 8;
61  if ( *c & bitf[i] )
62  * ( target + j ) += ( 1U << ( r - 1 ) );
63  else
64  *has_data = 1;
65  d += 1;
66  r -= 1;
67  }
68  while ( r > 0 );
69  *bit0_offset += 8; // update bit0_offset
70  }
71  * ( target + nc ) = '\0';
72  return bit_length;
73 }

76  \fn size_t get_bits_as_char_array ( char *target, uint8_t *has_data, uint8_t *source, size_t *bit0_offset, size_t bit_length )
77  \brief get a sequence of bits in data section 4 from a BUFR reports to get an array of chars
78  \param target string with the resulting array
79  \param has_data if 1 then has data, if 0 what we got is missing data
80  \param source buffer of uint8_t which is actually the data section 4.
81  \param bit0_offset bit offset of first bit of first char
82  \param bit_length number of bits to extract. Obviously this will be divisible by 8
83 
84  If returns the amount of bits readed. 0 if problems. It also update bit0_offset with the new bits.
85 */
86 size_t get_bits_as_char_array ( char *target, uint8_t *has_data, uint8_t *source, size_t *bit0_offset, size_t bit_length )
87 {
88  size_t i, j, k;
89  size_t nc;
90  uint8_t *c;
91 
92  if ( bit_length % 8 )
93  return 0; // bit_length needs to be divisible by 8
94 
95  //printf("bit_length=%lu\n", bit_length);
96 
97  nc = bit_length / 8;
98  i = ( *bit0_offset ) % 8;
99  k = 8 - i;
100  *has_data = 0; // marc if no missing data is present
101  for ( j = 0; j < nc ; j++ )
102  {
103  c = source + ( *bit0_offset ) / 8;
104  * ( target + j ) = ( *c & biti[i] );
105  if ( i )
106  {
107  * ( target + j ) <<= i;
108  * ( target + j ) |= ( ( * ( c + 1 ) & bitk[i - 1] ) >> k );
109  }
110  if ( * ( target + j ) != -1 )
111  *has_data = 1;
112  //printf("%c", * ( target + j ) );
113  *bit0_offset += 8; // update bit0_offset
114  }
115  * ( target + nc ) = '\0';
116  return bit_length;
117 }

171  \param bit0_offset Bit offset
172  \param bit_length Lenght (in bits) for the chunck to extract
173 
174  If bil_length is less than 8 then it uses version 2 \ref get_bits_as_uint32_t2
175 
176  If returns the amount of bits readed. 0 if problems. It also update bits_offset with the new bits.
177 */
178 size_t get_bits_as_uint32_t ( uint32_t *target, uint8_t *has_data, uint8_t *source, size_t *bit0_offset, size_t bit_length )
179 {
180  int i;
181  uint8_t *c;
182  uint64_t x;
183 
184  if ( bit_length > 32 || bit_length == 0 )
185  return 0;
186 
187  if ( bit_length < 8 )
188  return get_bits_as_uint32_t2 ( target, has_data, source, bit0_offset, bit_length );
189 
190  *target = 0;
191  *has_data = 0; // marc if no missing data is present
192  c = source + ( *bit0_offset ) / 8;
193  i = ( *bit0_offset ) % 8;
194  x = ( ( uint64_t ) ( *c & biti[i] ) << 32 ) + ( ( uint64_t ) ( * ( c + 1 ) ) << 24 ) + ( ( uint64_t ) ( * ( c + 2 ) ) << 16 ) +
195  ( ( uint64_t ) ( * ( c + 3 ) ) << 8 ) + ( uint64_t ) ( * ( c + 4 ) ); // 40 - i bits
196  x >>= ( 40 - i - bit_length );
197  *target = ( uint32_t ) x;
198  if ( ( 1UL << bit_length ) != ( x + 1UL ) )
199  *has_data = 1;
200 
201  *bit0_offset += bit_length; // update bit0_offset
202  return bit_length;
203 }

207  \fn int get_table_b_reference_from_uint32_t ( int32_t *target, uint8_t bits, uint32_t source )
208  \brief Get an int32_t from bits according with bufr criteria to change the reference of a descritor. Most significant bit in source is sign
209  \param target int32_t as result
210  \param bits number of bits to consider
211  \param source uint32_T with the data to transform
212 
213  If success return 0, 1 otherwise
214 */
215 // most significant of bits is the sign
216 int get_table_b_reference_from_uint32_t ( int32_t *target, uint8_t bits, uint32_t source )
217 {
218  uint32_t mask = 1;
219  if ( bits > 32 || bits == 0 )
220  return 1;
221 
222  if ( bits > 1 )
223  mask = ( ( uint32_t ) 1 << ( bits - 1 ) );
224 
225  if ( mask & source )
226  {
227  // case of negative number
228  *target = - ( int32_t ) ( source - mask );
229  }
230  else
231  *target = ( int32_t ) ( source );
232  return 0;
233 }
