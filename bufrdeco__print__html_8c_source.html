156/*!
157 \fn char * bufrdeco_print_atom_data_html ( char *target, struct bufr_atom_data *a )
158 \brief print the data in a struct \ref bufr_atom_data to a string as cells of table rows
159 \param target string where to print the result
160 \param a pointer to struct ref \ref bufr_atom_data with data to print
161 \param ss index of subset in bufr
162
163 Returns a pointer to result string
164*/
165char * bufrdeco_print_atom_data_html ( char *target, struct bufr_atom_data *a, uint32_t ss )
166{
167 char aux[256], *c;
168
169 c = target;
170 c += sprintf ( c, "<td class='desc'>%u %02u %03u</td>", a->desc.f, a->desc.x, a->desc.y );
171 c += sprintf ( c, "<td class='name'>%s</td>", a->name );
172 c += sprintf ( c, "<td class='unit'>%s</td>", a->unit );
173 if ( a->mask & DESCRIPTOR_VALUE_MISSING )
174 {
175 if ( a->is_bitmaped_by != 0 )
176 c += sprintf ( c, "<td class='missing'>MISSING</td><td>NOTE: Bitmaped by <a href='#d%u_%u'>#%u</a></td>", ss, a->is_bitmaped_by, a->is_bitmaped_by );
177 else if ( a->bitmap_to != 0 )
178 c += sprintf ( c, "<td class='missing'>MISSING</td><td>NOTE: Bitmap to <a href='#d%u_%u'>#%u</a></td>", ss, a->bitmap_to, a->bitmap_to );
179 else if ( a->related_to != 0 )
180 c += sprintf ( c, "<td class='missing'>MISSING</td><td>NOTE: Related to <a href='#d%u_%u'>#%u</a></td>", ss, a->related_to, a->related_to );
181 else
182 c += sprintf ( c, "<td class='missing'>MISSING</td><td></td>" );
183 }
184 else
185 {
186 if ( a->mask & DESCRIPTOR_HAVE_STRING_VALUE )
187 {
188 if ( a->is_bitmaped_by != 0 )
189 c += sprintf ( c, "<td></td><td class='cval'>%s<br>NOTE: Bitmaped by <a href='#d%u_%u'>#%u</a></td>\n", a->cval, ss, a->is_bitmaped_by, a->is_bitmaped_by );
190 else if ( a->bitmap_to != 0 )
191 c += sprintf ( c, "<td></td><td class='cval'>%s<br>NOTE: Bitmap to <a href='#d%u_%u'>#%u</a></td>\n", a->cval, ss, a->bitmap_to, a->bitmap_to );
192 else if ( a->related_to != 0 )
193 c += sprintf ( c, "<td></td><td class='cval'>%s<br>NOTE: Related to <a href='#d%u_%u'>#%u</a></td>\n", a->cval, ss, a->related_to, a->related_to );
194 else
195 c += sprintf ( c, "<td></td><td class='cval'>%s</td>\n", a->cval );
196 }
197 else if ( a->mask & DESCRIPTOR_HAVE_CODE_TABLE_STRING
198 || strstr ( a->unit, "CODE TABLE" ) == a->unit
199 || strstr ( a->unit, "Code table" ) == a->unit )
200 {
201 c += sprintf ( c, "<td class='ival'>%17u</td>", ( uint32_t ) a->val );
202 if ( a->is_bitmaped_by != 0 )
203 c += sprintf ( c, "<td class='ctable'>%s<br>NOTE: Bitmaped by <a href='#d%u_%u'>#%u</a></td>\n", a->ctable, ss, a->is_bitmaped_by, a->is_bitmaped_by );
204 else if ( a->bitmap_to != 0 )
205 c += sprintf ( c, "<td class='ctable'>%s<br>NOTE: Bitmap to <a href='#d%u_%u'>#%u</a></td>\n", a->ctable, ss, a->bitmap_to, a->bitmap_to );
206 else if ( a->related_to != 0 )
207 c += sprintf ( c, "<td class='ctable'>%s<br>NOTE: Related to <a href='#d%u_%u'>#%u</a></td>\n", a->ctable, ss, a->related_to, a->related_to );
208 else
209 c += sprintf ( c, "<td class='ctable'>%s</td>\n", a->ctable );
210 }
211 else if ( a->mask & DESCRIPTOR_HAVE_FLAG_TABLE_STRING )
212 {
213 c += sprintf ( c, "<td class='hval'>0x%08X</td>", ( uint32_t ) a->val );
214 if ( a->is_bitmaped_by != 0 )
215 c += sprintf ( c, "<td class='ctable'>%s<br>NOTE: Bitmaped by <a href='#%du_%u'>#%u</a></td>\n", a->ctable, ss, a->is_bitmaped_by, a->is_bitmaped_by );
216 else if ( a->bitmap_to != 0 )
217 c += sprintf ( c, "<td class='ctable'>%s<br>NOTE: Bitmap to <a href='#d%u_%u'>#%u</a></td>\n", a->ctable, ss, a->bitmap_to, a->bitmap_to );
218 else if ( a->related_to != 0 )
219 c += sprintf ( c, "<td class='ctable'>%s<br>NOTE: Related to <a href='#d%u_%u'>#%u</a></td>\n", a->ctable, ss, a->related_to, a->related_to );
220 else
221 c += sprintf ( c, "<td class='ctable'>%s</td>\n", a->ctable );
222 }
223 else
224 {
225 if ( a->is_bitmaped_by != 0 )
226 c += sprintf ( c, "<td class='rval'>%s</td><td>NOTE: Bitmaped by <a href='#d%u_%u'>#%u</a></td>" , get_formatted_value_from_escale ( aux, a->escale, a->val ), ss, a->is_bitmaped_by, a->is_bitmaped_by );
227 else if ( a->bitmap_to != 0 )
228 c += sprintf ( c, "<td class='rval'>%s</td><td>NOTE: Bitmap to <a href='#d%u_%u'>#%u</a></td>" , get_formatted_value_from_escale ( aux, a->escale, a->val ) , ss, a->bitmap_to, a->bitmap_to );
229 else if ( a->related_to != 0 )
230 c += sprintf ( c, "<td class='rval'>%s</td><td>NOTE: Related to <a href='#d%u_%u'>#%u</a></td>" , get_formatted_value_from_escale ( aux, a->escale, a->val ), ss, a->related_to, a->related_to );
231 else
232 c += sprintf ( c, "<td class='rval'>%s</td><td></td>" , get_formatted_value_from_escale ( aux, a->escale, a->val ) );
233 }
234 }
235 return target;
