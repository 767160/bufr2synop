38/*!
39 \fn int read_args( int _argc, char * _argv[])
40 \brief read the arguments from stdio
41 \param _argc number of arguments passed
42 \param _argv array of arguments
43
44 Returns 1 if succcess, -1 othewise
45*/
46int read_args ( int _argc, char * _argv[] )
47{
48 int iopt;
49
50 // Default values
51 EXTRACT = 0;
52 /*
53 Read input options
54 */
55 while ( ( iopt = getopt ( _argc, _argv, "hi:X" ) ) !=-1 )
56 switch ( iopt )
57 {
58 case 'X':
59 EXTRACT = 1;
60 break;
61 case 'i':
62 if ( strlen ( optarg ) < 256 )
63 strcpy ( ENTRADA, optarg );
64 break;
65 case 'h':
66 default:
67 print_usage();
68 exit ( EXIT_SUCCESS );
69 }
70
71 if ( ENTRADA[0] == '\0' )
72 {
73 printf ( "read_args(): It is needed an input file. Use -i option\n" );
74 return -1;
75 }
76 return 1;
77}

80int main ( int argc, char *argv[] )
81{
82 size_t subset;
83 struct bufrdeco_subset_sequence_data *seq;
84

98 }
99 else if ( bufrdeco_read_bufr ( &BUFR, ENTRADA ) )
100 {
101 printf ( "%s", BUFR.error );
102 exit ( EXIT_FAILURE );
103 }
104
105 print_sec0_info ( &BUFR );
106 print_sec1_info ( &BUFR );
107 print_sec3_info ( &BUFR );
108 print_sec4_info ( &BUFR );
109
110 if ( bufrdeco_parse_tree ( &BUFR ) )
111 {
112 printf ( "# %s", BUFR.error );
113 bufrdeco_close ( &BUFR );
114 exit ( EXIT_FAILURE );
115 }
116
117 bufrdeco_print_tree ( &BUFR );
118
119 for ( subset = 0; subset < BUFR.sec3.subsets ; subset++ )
120 {
121 if ( ( seq = bufrdeco_get_subset_sequence_data ( &BUFR ) ) == NULL )
122 {
123 printf ( "# %s", BUFR.error );
124 bufrdeco_close ( &BUFR );
125 exit ( EXIT_FAILURE );
126 }
127 bufrdeco_print_subset_sequence_data ( seq );
128 }
129
130 printf ( "So far so good !!\n" );
131 bufrdeco_close ( &BUFR );
132 exit ( EXIT_SUCCESS );
133}
