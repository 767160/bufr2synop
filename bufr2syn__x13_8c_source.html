26 /*!
27  \fn char * prec_to_RRR ( char *target, double r )
28  \brief converts a precipitation in Kg/m2 into a RRR string
29  \param r the precipitation
30  \param target the resulting string
31 */
32 char * prec_to_RRR ( char *target, double r )
33 {
34  if ( r == 0.0 )
35  strcpy ( target,"000" );
36  else if ( r < 0.1 )
37  strcpy ( target,"990" );
38  else if ( r < 1.0 )
39  sprintf ( target, "99%d", ( int ) ( r * 10.0 ) );
40  else
41  sprintf ( target, "%03d", ( int ) ( r ) );
42  return target;
43 
44 }
