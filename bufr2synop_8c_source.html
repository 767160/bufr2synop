131/*!
132 \fn int main(int argc, char *argv[])
133 \brief the standard C main entry function
134 \param argc Number of arguments entered in standard input, including the the name of this binary.
135 \param argv the array of strings with the arguents
136
137*/
138int main ( int argc, char *argv[] )
139{
140
141 int length; // max length of a bufrfile in bytes
142 int status = 0; // initial status
143 struct bufr_descriptor d; // bufr descriptor
144 unsigned int *kbuff;
145 char *c; // auxiliar pointer for sprintf tasks
146 int i, j, k, nsub, nsub1;
147 int kelem, kvals;
148 int icode = 0, ktdlen, ktdexl;
149 int current_ss;
150
151 if ( read_args ( argc, argv ) )
152 {
153 fprintf ( stderr, "%s: error geting arguments\n", SELF );
154 return ( EXIT_FAILURE );
155 }
156
157 // set needed enviroment before use bufr library
158 bufr_set_environment ( DEFAULT_BUFRTABLES, BUFRTABLES_DIR );
159
160 //bufrdeco_init_subset_sequence_data ( &SUBSET);
161 //memset(&SUBSET, 0, sizeof(struct bufr_subset_sequence_data));
162 SUBSET.sequence = DATARRAY;
163 SUBSET.nd = 0;
164 SUBSET.dim = BUFR_NMAXSEQ;
165
166 /**** Big loop. a cycle per file ****/
167 while ( get_bufrfile_path ( INPUTFILE, ERR ) )
168 {
169 kelem = KELEM;
170 kvals = KVALS;
171 length = BUFR_LEN;
172 if ( DEBUG )
173 printf ( "# Parsing file '%s'\n", INPUTFILE );
174
175 /* Try to guess a GTS header from filename*/
176 GTS_HEADER = guess_gts_header ( &HEADER, INPUTFILE );
177 if ( GTS_HEADER && DEBUG )
178 printf ( "#%s %s %s %s %s\n", HEADER.timestamp, HEADER.bname, HEADER.center, HEADER.dtrel, HEADER.order );
179
180 /* Read in bufr messages */
181 //status = readbufr ( fp, &BUFR_MESSAGE, &length );
182 if ( ( status = read_bufr ( BUFR_MESSAGE, INPUTFILE, &length ) ) )
183 {
184 if ( status == -1 )
185 fprintf ( stderr, "File '%s' do not begin with 'BUFR'.\n", INPUTFILE );
186 else if ( status == -2 )
187 fprintf ( stderr, "FILE '%s' do not end with '7777'.\n", INPUTFILE );
188 else if ( status == -3 )
189 fprintf ( stderr, "FILE '%s' has different size than expected.\n",
190 INPUTFILE );
191 else if ( status == -4 )
192 fprintf ( stderr, "Too small input FILE.\n" );
193 exit ( EXIT_FAILURE );
194 }
195 else if ( DEBUG )
196 {
197 printf ( "#It is OK.\n" );
198 printf ( "#message read " );
199 printf ( "%d\n", length );
200 printf ( "#%s\n", &BUFR_MESSAGE[0] ); // print the string 'BUFR'
201 }
202
203 /* Fortran routines works with 4-bytes words, so the readed file byte-oriented
204 is redefined as an array of unsigned int
205 */
206 kbuff = ( unsigned int * ) BUFR_MESSAGE;
207 length /= 4; // Now length is sized in words
208
209 if ( SHOW_ECMWF_OUTPUT )
210 {
211
212 bus012_ ( &length, kbuff, KSUP, KSEC0, KSEC1, KSEC2, &KERR );
213 buprs0_ ( KSEC0 );
214 buprs1_ ( KSEC1 );
215
216 // KSUP[5] is the number of subsets
217 if ( KSUP[5] > 1 )
218 kelem = kvals / KSUP[5];
219 else
220 kelem = KELEM;
221
222 if ( kelem > KELEM )
223 kelem = KELEM;
224 }
225
226 /* Here we call to bufrex() rotine, doing all de decode bufr work
227
228 ON INPUT:
229
230 length - An INTEGER variable containing length of Bufr message in words.
231 kbuff - An INTEGER array containing Bufr message.
232 kelem - An INTEGER variable containing expected number of expanded elements
233 kvals - An INTEGER variable containing expected number of data values.
234
235 ON OUTPUT:
236
237 KSEC0 - An INTEGER array (size 3) containing Bufr Section 0 information.
238 [0] Length of section 0 in bytes
239 [1] Total length of Bufr message in bytes
240 [2] Bufr Edition number (currently 4)
241
242 KSEC1 - An INTEGER array of at least 40 words containing Bufr Section 1 information. When Section
243 1 contains data for local use, KSEC1 should be sized accordingly.
244 [0] Length of section 1 in bytes
245 [1] Bufr Edition number (currently 4)
246 [2] Originating centre
247 [3] Update sequence number
248 [4] Flag (presence of Section 2 in the message)
249 [5] Bufr message type ( Bufr Table A)
250 [6] Bufr message subtype (local use)
251 [7] Version number of local table used
252 [8] Year
253 [9] Month
254 [10] Day
255 [11] Hour
256 [12] Minute
257 [13] Bufr Master Table used (zero if standard WMO FM 94-IX BUFR tables are used)
258 [14] Version number of Master table used
259 [15] Originating sub-centre
260 [16] International sub-category
261 [17] Second
262 [18-39] Local ADP centre information (byte by byte)
263
264 KSEC2 - An INTEGER array of 4096 words containing Bufr Section 2 information. ECMWF uses this
265 section to store Report Data Base KEY.
266 [0] Length of Section 2 in bytes
267 [1-4095] Report Data Base KEY in packed form
268
269 KSEC3 - An INTEGER array of 4 words containing Bufr Section 3 information.
270 [0] Length of Section 3 in bytes
271 [1] Reserved
272 [2] Number of subsets
273 [3] Flag (data type, compression)
274
275 KSEC4 - An INTEGER array of 2 words containing Section 4 information.
276 [0] Length of Section 4 in bytes
277 [1] Reserved
278
279 KSUP - An INTEGER array (size 9) containing supplementary information.
280 [0] Dimension of KSEC1 array
281 [1] Dimension of KSEC2 array
282 [2] Dimension of KSEC3 array
283 [3] Dimension of KSEC4 array
284 [4] Real number of expanded elements
285 [5] Number of subsets
286 [6] Real number of elements in CVALS array
287 [7] Total Bufr message length in bytes
288 [8] Dimension of KSEC0 array
289
290 CNAMES - CHARACTER*64 array of KELEM words containing element names.
291 CUNITS - CHARACTER*24 array of KELEM words containing element units.
292 VALUES - REAL*8 array of KVALS words containing element values.
293 CVALS - CHARACTER*80 array of KVALS containing CCITT IA No.5 element entries.
294 KERR - An INTEGER containing an error code.
295
296 */
297
298 KERR = 0;
299 bufrex_ ( &length, ( int * ) kbuff, KSUP, KSEC0, KSEC1, KSEC2, KSEC3, KSEC4,
300 &kelem, ( char ** ) CNAMES, ( char ** ) CUNITS, &kvals, VALUES,
301 ( char ** ) CVALS, &KERR );
302
303 if ( KERR )
304 {
305 printf ( "KERR=%d\n", KERR );
306 KERR = 0;
307 }
308
309 // Check about the WMO master table
310 if ( KSEC1[13] )
311 {
312 fprintf ( stderr,"Sorry, we only accept WMO BUFR master Table. sss = %d instead of 0\n" , KSEC1[13] );
313 exit ( EXIT_FAILURE );
314 }
315
316 if ( KSEC1[5] != 0 && KSEC1[5] != 1 && KSEC1[5] != 2 && KSEC1[5] != 7 )
317 {
318 fprintf ( stderr,"The data category %d is not parsed at the moment" , KSEC1[5] );
319 exit ( EXIT_FAILURE );
320 }
321
322 if ( KSUP[5] > KSUBS )
323 {
324 fprintf ( stderr,"%s: Warning, not all the %d subsets for this file can be managed, just %d\n", SELF, KSUP[5], KSUBS );
325 }
326
327 /*
328 Expand the descriptors
329
330 kdelen - A pointer to an INTEGER variable containing number of data descriptors in KTDLST array
331 KTDLST - An INTEGER array containing the list of kdtlen data descriptors
332 kdtexl - A pointer to an INTEGER variable containing number of expanded data descriptors
333 KTDEXP - An INTEGER array containing the list of KTDEXL data descriptors
334 KERR - An INTEGER containing error code.
335 */
336 busel_ ( &ktdlen, KTDLST, &ktdexl, KTDEXP, &KERR );
337
338 /*
339 read table C if VERBOSE
340 */
341 if ( VERBOSE )
342 {
343 if ( read_table_c ( TABLEC, &NLINES_TABLEC, BUFRTABLES_DIR, KSEC1 ) == 0 )
344 {
345 fprintf ( stderr,"Cannot read C Table File\n" );
346 exit ( EXIT_FAILURE );
347 }
348 }
349
350 if ( DEBUG )
351 {
352 for ( i = 0; i < ktdlen; i++ )
353 {
354 integer_to_descriptor ( &d, KTDLST[i] );
355 printf ( "#KTDLST[%03d]=%d %02d %03d\n", i, d.f, d.x, d.y );
356 }
357 }
358
359 // loop for every subset
360 for ( nsub = 0; nsub < KSUP[5]; nsub++ )
361 {
362 nsub1 = nsub + 1;
363
364 // clean sequence
365 //memset ( &SUBSET, 0, sizeof ( struct bufr_subset_sequence_data ) ); /***/
366 memset ( SUBSET.sequence, 0, ( sizeof ( struct bufr_atom_data ) ) * BUFR_NMAXSEQ );
367 SUBSET.nd = 0;
368
369 // clean REPORT
370 memset ( &REPORT, 0, sizeof ( struct metreport ) );
371
372 // sets GTS header, common for all subsets in a bufr file
373 if ( GTS_HEADER )
374 REPORT.h = &HEADER;
375
376 /*
377 Expand the descriptors for a subset
378
379 NOTE that not all subset have the same descriptor expanded list because of replication factor
380 so is safer expand the subset descriptors individually
381
382 Also NOTE that all integers must be called by reference
383 */
384 busel2_ ( &nsub1, &kelem, &ktdlen, ( char ** ) KTDLST, &ktdexl, ( char ** ) KTDEXP, ( char ** ) CNAMES, ( char ** ) CUNITS, &KERR );
385
386 if ( KERR )
387 continue;
388
389 for ( j = 0 ; j < ktdexl && j < KSUBS ; j++ )
390 {
391 i = nsub * KELEM + j;
392 LINAUX[0] = '\0'; // clean the output line
393 c = LINAUX;
394 charray_to_string ( SUBSET.sequence[j].name, ( unsigned char * ) CNAMES[j], 64 );
395 charray_to_string ( SUBSET.sequence[j].unit, ( unsigned char * ) CUNITS[j], 24 );
396 integer_to_descriptor ( &SUBSET.sequence[j].desc, KTDEXP[j] );
397
398 c += sprintf ( c, "KTDEXP[%03d]=%s |%03d |", j, SUBSET.sequence[j].desc.c, nsub );
399 c += sprintf ( c, "'%s'|", SUBSET.sequence[j].name );
400 if ( VALUES[i] != MISSING_REAL )
401 {
402 if ( strstr ( SUBSET.sequence[j].unit,"CCITTIA5" ) != NULL )
403 {
404 SUBSET.sequence[j].mask |= DESCRIPTOR_HAVE_STRING_VALUE;
405 k = ( ( int ) VALUES[i] ) % 1000;
406 charray_to_string ( SUBSET.sequence[j].cval, ( unsigned char * ) CVALS[ ( int ) ( VALUES[i]/1000.0 ) - 1], k );
407 c += sprintf ( c, "'%s'", SUBSET.sequence[j].cval );
408 }
409 else if ( strstr ( SUBSET.sequence[j].unit,"CODE TABLE" ) == SUBSET.sequence[j].unit )
410 {
411 SUBSET.sequence[j].mask |= DESCRIPTOR_IS_CODE_TABLE;
412 SUBSET.sequence[j].val = VALUES[i];
413 if ( get_explained_table_val ( SUBSET.sequence[j].ctable, 256, TABLEC, NLINES_TABLEC, &SUBSET.sequence[j].desc, ( int ) VALUES[i] ) != NULL )
414 {
415 SUBSET.sequence[j].mask |= DESCRIPTOR_HAVE_CODE_TABLE_STRING;
416 if ( VERBOSE )
417 c += sprintf ( c, "%9d : '%s'", ( int ) VALUES[i], SUBSET.sequence[j].ctable );
418 }
419 else if ( VERBOSE )
420 {
421 c += sprintf ( c,"%9d : 'NOT FOUND'", ( int ) VALUES[i] );
422 }
423 else
424 c += sprintf ( c, "%9d", ( int ) VALUES[i] );
425 }
426 else if ( strstr ( SUBSET.sequence[j].unit,"FLAG" ) == SUBSET.sequence[j].unit )
427 {
428 SUBSET.sequence[j].mask |= DESCRIPTOR_IS_FLAG_TABLE;
429 SUBSET.sequence[j].val = VALUES[i];
430 if ( get_explained_flag_val ( SUBSET.sequence[j].ctable, 256, TABLEC, NLINES_TABLEC, &SUBSET.sequence[j].desc, ( unsigned long ) VALUES[i] ) != NULL )
431 {
432 SUBSET.sequence[j].mask |= DESCRIPTOR_HAVE_FLAG_TABLE_STRING;
433 if ( VERBOSE )
434 c += sprintf ( c, "%9d : '%s'", ( int ) VALUES[i], SUBSET.sequence[j].ctable );
435 }
436 else if ( VERBOSE )
437 {
438 c += sprintf ( c, "%9d : 'NOT FOUND'", ( int ) VALUES[i] );
439 }
440 else
441 c += sprintf ( c, "%9d", ( int ) VALUES[i] );
442 }
443 else
444 {
445 SUBSET.sequence[j].mask |= DESCRIPTOR_HAVE_REAL_VALUE;
446 SUBSET.sequence[j].val = VALUES[i];
447 c += sprintf ( c, "%14.4lf %s ", VALUES[i], SUBSET.sequence[j].unit );
448 }
449 }
450 else
451 {
452 SUBSET.sequence[j].mask |= DESCRIPTOR_VALUE_MISSING;
453 c += sprintf ( c, " MISSING %s ", SUBSET.sequence[j].unit );
454 }
455 SUBSET.nd = j;
456 c += sprintf ( c, "\n" );
457 if ( SHOW_SEQUENCE && strlen ( LINAUX ) )
458 printf ( "#%s", LINAUX );
459 }
460
461 /**** the call to parse the sequence and transform to solicited asciicode, if possible *****/
462 if ( parse_subset_sequence ( &REPORT, &SUBSET, &STATE, KTDLST, ktdlen, KSEC1, ERR ) )
463 {
464 if ( DEBUG )
465 fprintf ( stderr, "#%s\n", ERR );
466 }
467 else if ( XML )
468 {
469 if ( nsub == 0 )
470 fprintf ( stdout, "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n" );
471 print_xml ( stdout, &REPORT );
472 }
473 else if ( JSON )
474 {
475 print_json ( stdout, &REPORT );
476 }
477 else if ( CSV )
478 {
479 if ( nsub == 0 )
480 fprintf ( stdout, "TYPE,FILE,DATETIME,INDEX,NAME,COUNTRY,LATITUDE,LONGITUDE,ALTITUDE,REPORT\n" );
481 print_csv ( stdout, &REPORT );
482 }
483 else
484 {
485 print_plain ( stdout, &REPORT );
486 }
487 }
488
489
490 if ( SHOW_ECMWF_OUTPUT )
491 {
492 /* Section 2 of the Bufr message is an optional section and every ADP centre can pack
493 any information in this section. The Bufr software decodes this local information
494 and stores it into KSEC2 array. ECMWF is storing RDB key in the Section 2 of the Bufr
495 messages. To print content of the Section 2, subroutine BUUKEY must be called before
496 the BUPRS2 routine. */
497 if ( KSEC1[4] )
498 buukey_ ( KSEC1, KSEC2, KEY, KSUP, &KERR );
499
500 /* Returns lists of unexpanded and expanded data descriptors from the Bufr message.
501 The lists contains Bufr Table D sequence numbers, and the Bufr Table B reference numbers. */
502
503 busel_ ( &ktdlen, KTDLST, &ktdexl, KTDEXP, &KERR );
504
505 /* Prints section 3.
506 Prior to calling the BUPRS3 routine, the BUSEL or BUSEL2 routine has to be called to get lists
507 of unexpanded and fully expanded Data descriptors. In the case of multi-subset uncompressed bufr
508 data the expanded list of descriptors might be different for different subsets. */
509 buprs3_ ( KSEC3, &ktdlen, KTDLST, &ktdexl, KTDEXP, &kelem,
510 ( char ** ) CNAMES );
511
512 /* Print data */
513 icode = 0;
514 current_ss = 1;
515 buprt_ ( &icode, &current_ss, &KSEC3[2], &kelem, ( char ** ) CNAMES,
516 ( char ** ) CUNITS, ( char ** ) CVALS, &kvals, VALUES, KSUP, KSEC1,
517 &KERR );
518 }
519
520 if ( DEBUG )
521 {
522 for ( i = 0; i < 3 ; i++ )
523 printf ( "#KSEC0[%d] = %d\n", i, KSEC0[i] );
524 for ( i = 0; i < 40 ; i++ )
525 printf ( "#KSEC1[%d] = %d\n", i, KSEC1[i] );
526 if ( KSEC1[4] )
527 {
528 for ( i = 0; i < 4096; i++ )
529 printf ( "#KSEC2[%d] = %d\n", i, KSEC2[i] );
530 }
531 for ( i = 0; i < 4; i++ )
532 printf ( "#KSEC3[%d] = %d\n", i, KSEC3[i] );
533
534 for ( i = 0; i < 9; i++ )
535 printf ( "#KSUP[%d] = %d\n", i, KSUP[i] );
536 }
537
538 NFILES ++;
539 } // End of big loop parsing files
540
541 return KERR;
542
543}
