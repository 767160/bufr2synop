26 /*!
27  \fn int syn_parse_x01 ( struct synop_chunks *syn, struct bufr2tac_subset_state *s )
28  \brief Parse a expanded descriptor with X = 01
29  \param syn pointer to a struct \ref synop_chunks where to set the results
30  \param s pointer to a struct \ref bufr2tac_subset_state where is stored needed information in sequential analysis
31 
32  It returns 0 if success, 1 if problems when processing. If a descriptor is not processed returns 0 anyway
33 */
34 int syn_parse_x01 ( struct synop_chunks *syn, struct bufr2tac_subset_state *s )
35 {
36  char aux[80];
37 
38  if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
39  return 0;
40 
41  switch ( s->a->desc.y )
42  {
43  case 1: // 0 01 001 . WMO block number
44  sprintf ( syn->s0.II, "%02d", s->ival );
45  break;
46  case 2: // 0 01 002 . WMO station number
47  sprintf ( syn->s0.iii, "%03d", s->ival );
48  break;
49  case 3: // 0 01 003 . WMO Region
50  sprintf ( syn->s0.A1, "%d", s->ival );
51  if ( strcmp ( syn->s0.A1, "1" ) == 0 )
52  strcpy ( syn->s0.Reg, "I" );
53  else if ( strcmp ( syn->s0.A1, "2" ) == 0 )
54  strcpy ( syn->s0.Reg, "II" );
55  else if ( strcmp ( syn->s0.A1, "3" ) == 0 )
56  strcpy ( syn->s0.Reg, "III" );
57  else if ( strcmp ( syn->s0.A1, "4" ) == 0 )
58  strcpy ( syn->s0.Reg, "IV" );
59  else if ( strcmp ( syn->s0.A1, "5" ) == 0 )
60  strcpy ( syn->s0.Reg, "V" );
61  else if ( strcmp ( syn->s0.A1, "6" ) == 0 )
62  strcpy ( syn->s0.Reg, "VI" );
63  break;
64  case 4: // 0 01 004 . WMO Subarea
65  case 20: // 0 01 020 . WMO region subarea
66  sprintf ( syn->s0.bw, "%d", s->ival );
67  break;
68  case 11: // 0 01 011. Ship or mobile land station index
69  if ( strlen ( s->a->cval ) < 16 )
70  {
71  strcpy ( aux, s->a->cval );
72  adjust_string ( aux );
73  if ( aux[0] == '\xff' )
74  break;
75  if ( strlen ( aux ) < 10 )
76  strcpy ( syn->s0.D_D, aux );
77  }
78  break;
79  case 15: // 0 01 015 . Station or site name
80  case 18: // 0 01 018 . Short station or site name
81  case 19: // 0 01 019 . Long station or site name
82  if ( strlen ( s->a->cval ) <= 80 )
83  {
84  strcpy ( aux, s->a->cval );
85  adjust_string ( aux );
86  strcpy ( s->name, aux );
87  s->mask |= SUBSET_MASK_HAVE_NAME;
88  }
89  break;
90  case 101: // 0 01 101 . State identifier
91  if ( strlen ( s->a->ctable ) <= 256 )
92  {
93  strcpy ( aux, s->a->ctable );
94  adjust_string ( aux );
95  strcpy ( s->country, aux );
96  s->mask |= SUBSET_MASK_HAVE_COUNTRY;
97  }
98  break;
99  default:
100  break;
101  }
102 
103  //set WMO region if we know II and iii and still no set A1
104  if ((syn->s0.A1[0] == 0) && syn->s0.II[0] && syn->s0.iii[0])
105  {
106  guess_WMO_region ( syn->s0.A1, syn->s0.Reg, syn->s0.II, syn->s0.iii );
107  }
108  return 0;
109 }
