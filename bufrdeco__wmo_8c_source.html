136 /*!
137  \fn int bufr_read_tables_wmo (struct bufrdeco *b, char *tables_dir)
138  \brief Read the tables according with bufr file data from a bufr table directory
139  \param b basic struct with needed data
140  \param tables_dir complete path string for directory with WMOF bufr tables. Must be ended with a '/' . If NULL then is the default
141 
142  The default directories where to search bufr tables are stored in \ref DEFAULT_BUFRTABLES_WMO_CSV_DIR1 and \ref DEFAULT_BUFRTABLES_WMO_CSV_DIR2
143 */
144 int bufr_read_tables_wmo ( struct bufrdeco *b, char *tables_dir )
145 {
146 
147  // get tablenames
148  if ( get_wmo_tablenames ( b, tables_dir ) )
149  {
150  sprintf ( b->error, "bufrdeco_read_tables_ecmwf(): Cannot find bufr tebles\n" );
151  return 1;
152  }
153 
154  // If tables still not initialized then do it
155  if ( b->tables == NULL && bufrdeco_init_tables ( & ( b->tables ) ) )
156  {
157  sprintf ( b->error, "bufrdeco_read_tables_ecmwf(): Cannot allocate memory for tables\n" );
158  return 1;
159  }
160 
161  // And now read tables
162  if ( bufr_read_tableb_csv ( & ( b->tables->b ), b->error ) )
163  {
164  return 1;
165  }
166 
167  if ( bufr_read_tablec_csv ( & ( b->tables->c ), b->error ) )
168  {
169  return 1;
170  }
171 
172  if ( bufr_read_tabled_csv ( & ( b->tables->d ), b->error ) )
173  {
174  return 1;
175  }
176  return 0;
177 }
