32/*!

34 \brief Get the complete pathnames for WMO csv table files needed by a bufr message
35 \param b pointer for a struct \ref bufrdeco

37 For WMO files this format is adopted
38 BUFR_XX_Y_Z_TableB_en for table B
39 BUFR_XX_Y_Z_CodeFlag for Code table and Flag table. This is equivalent to table C in ECMWF package
40 BUFR_XX_Y_Z_TableD_en for table D
41
42 XX is the Version number of master table used

44 Z is minor revision (currently ignored)

151/*!
152 \fn int bufr_read_tables_wmo (struct bufrdeco *b)
153 \brief Read the tables according with bufr file data from a bufr table directory
154 \param b basic struct with needed data
155
156 The default directories where to search bufr tables are stored in \ref DEFAULT_BUFRTABLES_WMO_CSV_DIR1 and \ref DEFAULT_BUFRTABLES_WMO_CSV_DIR2
157*/
158int bufr_read_tables_wmo ( struct bufrdeco *b )
159{
160
161 // get tablenames
162 if ( get_wmo_tablenames ( b ) )
163 {
164 sprintf ( b->error, "bufrdeco_read_tables_ecmwf(): Cannot find bufr tables\n" );
165 return 1;
166 }
167
168 // If tables still not initialized then do it
169 if ( b->tables == NULL && bufrdeco_init_tables ( & ( b->tables ) ) )
170 {
171 sprintf ( b->error, "bufrdeco_read_tables_ecmwf(): Cannot allocate memory for tables\n" );
172 return 1;
173 }
174
175 // And now read tables
176 if ( bufr_read_tableb_csv ( & ( b->tables->b ), b->error ) )
177 {
178 return 1;
179 }
180
181 if ( bufr_read_tablec_csv ( & ( b->tables->c ), b->error ) )
182 {
183 return 1;
184 }
185
186 if ( bufr_read_tabled_csv ( & ( b->tables->d ), b->error ) )
187 {
188 return 1;
189 }
190 return 0;
191}
