68 {

70 
71  if ( s->a->mask & DESCRIPTOR_VALUE_MISSING)

76  case 1: // 0 06 001

84  s->lon = s->val;

86  default:

88  }
89 
90  // check if set both LaLaLa and LoLoLoLo to set Qc
91  if ((syn->s0.Qc[0] == 0) && syn->s0.LaLaLa[0] && syn->s0.LoLoLoLo[0])
92  {
93  if (s->mask & SUBSET_MASK_LATITUDE_SOUTH)
94  {
95  if (s->mask & SUBSET_MASK_LONGITUDE_WEST)
96  strcpy(syn->s0.Qc, "5");
97  else
98  strcpy(syn->s0.Qc, "3");
99  }
100  else

104  else
105  strcpy(syn->s0.Qc, "1");
106  }
107  }
108 
109  // check if about MMM

111  {
112  latlon_to_MMM(syn->s0.MMM, s->lat, s->lon);
113  }
