109int main ( int argc, char *argv[] )
110{
111
112 FILE *fp;
113 char filename[256];
114 int length=512000;
115 int status = 0;
116
117 unsigned int *kbuff;
118 int ksup[9];
119 int ksec0[3];
120 int ksec1[40];
121 int ksec2[4096];
122 int ksec3[4];
123 int ksec4[2];
124 int key[46];
125 int kerr;
126
127 int kelem = KELEM, kvals = KVALS;
128 int icode = 0, ktdlen, ktdexl;
129 int current_ss;
130
131 /* Get input and output file name. */
132 /* ------------------------------- */
133
134 if ( argc != 3 )
135 {
136 printf ( "Usage: bufr_decode -i infile \n" );
137 printf ( "Please try again. \n" );
138 exit ( 1 );
139 }
140
141 printf ( "%c",argc );
142
143 if ( !strcmp ( argv[1],"-i" ) ) strcpy ( filename,argv[2] );
144 else
145 {
146 printf ( "Usage: bufr_decode -i infile \n" );
147 exit ( 1 );
148 }
149
150 /* Open input file */
151 /* --------------- */
152
153 if ( ( fp = fopen ( filename,"r" ) ) == NULL )
154 {
155 printf ( "cannot open file\n" );
156 exit ( 1 );
157 }
158
159 /* Read in bufr messages */
160 /* --------------------- */
161
162 while ( status >= 0 )
163 {
164 status = readbufr ( fp,&bufr_message,&length );
165
166 if ( status == -1 ) printf ( "End of file.\n" );
167 else if ( status == -2 ) printf ( "Error in file handling\n" );
168 else if ( status == -3 ) printf ( "Too small input array.\n" );
169 else if ( status == -4 ) printf ( "Too small input array.\n" );
170 else
171 {
172 printf ( "It is OK.\n" );
173 printf ( "message read " );
174 printf ( "%d\n",length );
175 printf ( "%s\n",&bufr_message[0] );
176 }
177 status=-1;
178
179 /* Expand bufr message calling fortran program */
180 kbuff = ( unsigned int * ) bufr_message;
181 length /= 4; // Now length is sized in words
182
183 bus012_ ( &length, kbuff , ksup, ksec0, ksec1, ksec2, &kerr ) ;
184 buprs0_ ( ksec0 );
185 buprs1_ ( ksec1 );
186
187 if ( ksup[5] > 1 )
188 kelem = kvals/ksup[5];
189 else
190 kelem = KELEM;
191
192 if ( kelem > KELEM ) kelem = KELEM;
193 kerr = 0;
194 bufrex_ ( &length, ( int * ) kbuff,ksup,ksec0,ksec1,ksec2,ksec3,ksec4,
195 &kelem, ( char ** ) cnames, ( char ** ) cunits,&kvals,
196 values, ( char ** ) cvals,&kerr );
197 if ( kerr )
198 {
199 kerr = 0;
200 }
201
202 /* NOTE the differences between array indexes in fortran and C
203 so fortran KSEC3[3] is C ksec3[2] */
204 printf ( "ksec3[3]=%d\n", ksec3[2] );
205 buukey_ ( ksec1,ksec2,key,ksup,&kerr );
206
207 busel_ ( &ktdlen,ktdlst,&ktdexl,ktdexp,&kerr );
208 buprs3_ ( ksec3,&ktdlen,ktdlst,&ktdexl,ktdexp,&kelem, ( char ** ) cnames );
209
210 icode = 0;
211 current_ss = 1;
212 buprt_ ( &icode,&current_ss,&ksec3[2],&kelem, ( char ** ) cnames,
213 ( char ** ) cunits, ( char ** ) cvals,
214 &kvals,values,ksup,ksec1,&kerr );
215
216 }
217
218 return kerr;
219
220}
