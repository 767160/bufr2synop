386 /*!
387  \fn int print_buoy(char *report, size_t lmax, struct buoy_chunks *b)
388  \brief prints a buoy into a string
389  \param report target string
390  \param lmax max size of string
391  \param b pointer to a struct \ref buoy_chunks with the result of parse tasks
392 
393  returns 0 if all went right
394 */
395 int print_buoy ( char *report, size_t lmax, struct buoy_chunks *b )
396 {
397  char *c;
398 
399  c = report;
400 
401  // Needs time extension
402  if ( b->e.YYYY[0] == 0 )
403  {
404  return 1;
405  }
406 
407  print_buoy_sec0 ( &c, lmax, b );
408 
409  if ( b->mask & ( BUOY_SEC1 | BUOY_SEC2 | BUOY_SEC3 ) )
410  {
411  print_buoy_sec1 ( &c, lmax - strlen ( report ), b );
412 
413  print_buoy_sec2 ( &c, lmax - strlen ( report ), b );
414 
415  print_buoy_sec3 ( &c, lmax - strlen ( report ), b );
416  }
417  else
418  {
419  c+= sprintf ( c, " NIL" );
420  }
421  c += sprintf ( c, "=" );
422 
423  return 0;
424 }
